```typescript
/** Pure rectangle intersection test — exported for testability. */
export function rectsIntersect(
  a: { x: number; y: number; w: number; h: number },
  b: { x: number; y: number; width: number; height: number }
): boolean {
  return (
    a.x < b.x + b.width &&
    a.x + a.w > b.x &&
    a.y < b.y + b.height &&
    a.y + a.h > b.y
  );
}

export class SuperGridBBoxCache {
  private _cache = new Map<string, DOMRect>();
  private _gridEl: HTMLElement | null = null;

  attach(gridEl: HTMLElement): void {
    this._gridEl = gridEl;
  }

  detach(): void {
    this._gridEl = null;
    this._cache.clear();
  }

  scheduleSnapshot(): void {
    requestAnimationFrame(() => this._snapshot());
  }

  getRect(cellKey: string): DOMRect | undefined {
    return this._cache.get(cellKey);
  }

  hitTest(lassoRect: { x: number; y: number; w: number; h: number }): string[] {
    const hits: string[] = [];
    for (const [key, rect] of this._cache) {
      if (rectsIntersect(lassoRect, rect)) hits.push(key);
    }
    return hits;
  }

  private _snapshot(): void {
    if (!this._gridEl) return;
    this._cache.clear();
    const cells = this._gridEl.querySelectorAll<HTMLElement>('.data-cell');
    for (const cell of cells) {
      const key = cell.dataset['key'];
      if (key) this._cache.set(key, cell.getBoundingClientRect());
    }
  }
}
```

**IMPORTANT — rectsIntersect must use `b.x + b.width` and `b.y + b.height` (not `b.right` / `b.bottom`) because jsdom DOMRect may not have right/bottom getters. The plain object form `{ x, y, width, height }` is safe for both browser DOMRect and test mocks.**

Since jsdom's `getBoundingClientRect()` returns zero rects, the test approach for hitTest should be:
1. Create `.data-cell` divs with `dataset.key` set
2. Manually populate the cache's internal Map using a test-only approach: either (a) expose `_snapshot` indirectly via `scheduleSnapshot` + flush rAF, or (b) call `scheduleSnapshot()`, flush rAF using `vi.useFakeTimers()` + `vi.advanceTimersToNextTimer()`, and mock `getBoundingClientRect` on the test elements before the snapshot fires.

Recommended: use `vi.spyOn(element, 'getBoundingClientRect').mockReturnValue(new DOMRect(x, y, w, h))` on each test `.data-cell` element, then trigger `scheduleSnapshot()` and advance the rAF timer.
