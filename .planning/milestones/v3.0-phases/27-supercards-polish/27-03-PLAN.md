       function mulberry32(seed: number): () => number {
         return function() {
           seed |= 0; seed = seed + 0x6D2B79F5 | 0;
           let t = Math.imul(seed ^ seed >>> 15, 1 | seed);
           t = t + Math.imul(t ^ t >>> 7, 61 | t) ^ t;
           return ((t ^ t >>> 14) >>> 0) / 4294967296;
         };
       }
       ```

       function makeSyntheticCells(cols: number, rows: number, seed = 42): Array<Record<string, unknown>> {
         const rand = mulberry32(seed);
         const cells: Array<Record<string, unknown>> = [];
         for (let r = 0; r < rows; r++) {
           for (let c = 0; c < cols; c++) {
             const count = Math.floor(rand() * 10) + 1;
             cells.push({
               col_field: `col_${c}`,
               row_field: `row_${r}`,
               count,
               card_ids: Array.from({ length: count }, (_, i) => `card-${r}-${c}-${i}`).join(','),
             });
           }
         }
         return cells;
       }
       ```

    **PLSH-01 test: 50x50 grid render < 16ms (tolerance 19.2ms)**

    ```
    describe('PLSH-01 — 50x50 grid render < 16ms', () => {
      const BUDGET_MS = 16;
      const TOLERANCE = 1.20;
      const LIMIT_MS = BUDGET_MS * TOLERANCE;  // 19.2ms

      it('_renderCells() on 50x50 cells completes within tolerance', () => {
        // Setup: create SuperGrid with mocks, mount it
        const root = document.createElement('div');
        document.body.appendChild(root);
        const grid = new SuperGrid(root, makeMockBridge(), makeMockProvider(
          [{ field: 'col_field' }], [{ field: 'row_field' }]
        ), makeMockFilter(), undefined, undefined, makeMockDensity('matrix'));
        grid.mount();

        // Generate 50x50 synthetic cells
        const cells = makeSyntheticCells(50, 50, 42);

        // Warm up: one render to JIT compile
        (grid as any)._lastCells = cells;
        (grid as any)._lastColAxes = [{ field: 'col_field' }];
        (grid as any)._lastRowAxes = [{ field: 'row_field' }];
        (grid as any)._renderCells(cells);

        // Measure: run N iterations, take p95
        const N = 20;
        const samples: number[] = [];
        for (let i = 0; i < N; i++) {
          const start = performance.now();
          (grid as any)._renderCells(cells);
          samples.push(performance.now() - start);
        }

        const p95 = computeP95(samples);
        expect(p95).toBeLessThan(LIMIT_MS);

        grid.destroy();
        document.body.removeChild(root);
      });
    });
    ```

    **PLSH-03 test: Axis transpose reflow < 300ms (tolerance 360ms)**

    This test measures the full reflow cycle: provider setter -> _fetchAndRender -> _renderCells + D3 transition.

    ```
    describe('PLSH-03 — Axis transpose reflow < 300ms', () => {
      const BUDGET_MS = 300;
      const TOLERANCE = 1.20;
      const LIMIT_MS = BUDGET_MS * TOLERANCE;  // 360ms

      it('setColAxes/setRowAxes -> _fetchAndRender -> _renderCells completes within tolerance', async () => {
        // Setup: SuperGrid with mocks, initial 10x10 grid
        const root = document.createElement('div');
        document.body.appendChild(root);
        const mockBridge = makeMockBridge();
        const cells10x10 = makeSyntheticCells(10, 10, 42);
        mockBridge.superGridQuery.mockResolvedValue(cells10x10);

        const provider = makeMockProvider(
          [{ field: 'col_field' }], [{ field: 'row_field' }]
        );
        const grid = new SuperGrid(root, mockBridge, provider, makeMockFilter(),
          undefined, undefined, makeMockDensity('matrix'));
        grid.mount();

        // Initial render
        await (grid as any)._fetchAndRender();

        // Prepare transposed cells
        const cellsTransposed = makeSyntheticCells(10, 10, 99);
        mockBridge.superGridQuery.mockResolvedValue(cellsTransposed);

        // Measure: transpose (swap axes) -> _fetchAndRender -> _renderCells
        const N = 10;
        const samples: number[] = [];
        for (let i = 0; i < N; i++) {
          // Alternate axis configuration to simulate transpose
          const colAxes = i % 2 === 0 ? [{ field: 'row_field' }] : [{ field: 'col_field' }];
          const rowAxes = i % 2 === 0 ? [{ field: 'col_field' }] : [{ field: 'row_field' }];
          provider.getStackedGroupBySQL.mockReturnValue({ colAxes, rowAxes });

          const start = performance.now();
          await (grid as any)._fetchAndRender();
          samples.push(performance.now() - start);
        }

        const p95 = computeP95(samples);
        expect(p95).toBeLessThan(LIMIT_MS);

        grid.destroy();
        document.body.removeChild(root);
      });
    });
    ```
