```typescript
export function renderDimensionCard(d: CardDatum): HTMLDivElement {
  const card = document.createElement('div');
  card.className = 'card';
  card.dataset['id'] = d.id;
  card.setAttribute('tabindex', '-1');

  // Audit data attributes (Phase 37 pattern)
  const changeStatus = auditState.getChangeStatus(d.id);
  if (changeStatus) card.dataset['audit'] = changeStatus;
  if (d.source) card.dataset['source'] = d.source;

  // Icon badge
  const iconEl = document.createElement('span');
  iconEl.className = 'card__icon';
  iconEl.textContent = CARD_TYPE_ICONS[d.card_type] ?? 'N';
  card.appendChild(iconEl);

  // Title
  const titleEl = document.createElement('span');
  titleEl.className = 'card__title';
  titleEl.textContent = d.name || 'Untitled';
  titleEl.title = d.name;
  card.appendChild(titleEl);

  // Content preview (hidden at 1x via CSS)
  const previewEl = document.createElement('div');
  previewEl.className = 'card__preview';
  previewEl.textContent = d.body_text ?? d.status ?? d.folder ?? '';
  card.appendChild(previewEl);

  // Tags placeholder (hidden at 1x/2x via CSS; populated by caller if tags available)
  const tagsEl = document.createElement('div');
  tagsEl.className = 'card__tags';
  card.appendChild(tagsEl);

  // Properties toggle + section (5x only, hidden by default)
  const propsToggle = document.createElement('button');
  propsToggle.className = 'card__props-toggle';
  propsToggle.textContent = 'Properties \u25BE';
  propsToggle.setAttribute('aria-expanded', 'false');
  card.appendChild(propsToggle);

  const propsEl = document.createElement('div');
  propsEl.className = 'card__props';
  card.appendChild(propsEl);

  // Wire properties toggle
  propsToggle.addEventListener('click', (e) => {
    e.stopPropagation();
    const isOpen = propsEl.classList.toggle('card__props--open');
    propsToggle.setAttribute('aria-expanded', String(isOpen));
    propsToggle.textContent = isOpen ? 'Properties \u25B4' : 'Properties \u25BE';
  });

  return card;
}
```

Keep the existing `renderHtmlCard()` and `renderSvgCard()` unchanged — they are still used by views not yet migrated.

const levels: Array<'1x' | '2x' | '5x'> = ['1x', '2x', '5x'];
for (const level of levels) {
  const btn = document.createElement('button');
  btn.className = 'dim-btn';
  btn.dataset['size'] = level;
  btn.textContent = level.replace('x', '\u00D7');  // 1x -> 1×
  btn.setAttribute('aria-pressed', level === this._currentDimension ? 'true' : 'false');
  btn.setAttribute('tabindex', level === this._currentDimension ? '0' : '-1');
  btn.addEventListener('click', () => this._setDimension(level));
  dimSwitcher.appendChild(btn);
}

// Roving tabindex: arrow keys navigate between buttons
dimSwitcher.addEventListener('keydown', (e: KeyboardEvent) => {
  const btns = Array.from(dimSwitcher.querySelectorAll<HTMLButtonElement>('.dim-btn'));
  const currentIdx = btns.findIndex(b => b.dataset['size'] === this._currentDimension);
  let nextIdx = currentIdx;
  if (e.key === 'ArrowRight' || e.key === 'ArrowDown') {
    e.preventDefault();
    nextIdx = Math.min(currentIdx + 1, btns.length - 1);
  } else if (e.key === 'ArrowLeft' || e.key === 'ArrowUp') {
    e.preventDefault();
    nextIdx = Math.max(currentIdx - 1, 0);
  }
  if (nextIdx !== currentIdx) {
    btns[currentIdx]!.setAttribute('tabindex', '-1');
    btns[nextIdx]!.setAttribute('tabindex', '0');
    btns[nextIdx]!.focus();
  }
});

dimSection.appendChild(dimSwitcher);
this._dimSwitcherEl = dimSection;
// Insert dimSection into the root element before the content area
this._rootEl.insertBefore(dimSection, this._contentEl);
```

4. Add `_setDimension(level)` private method:

private _setDimension(level: '1x' | '2x' | '5x'): void {
  if (level === this._currentDimension) return;
  this._currentDimension = level;
  // Update button states
  if (this._dimSwitcherEl) {
    const btns = this._dimSwitcherEl.querySelectorAll<HTMLButtonElement>('.dim-btn');
    for (const btn of btns) {
      const isActive = btn.dataset['size'] === level;
      btn.classList.toggle('dim-btn--active', isActive);
      btn.setAttribute('aria-pressed', String(isActive));
      btn.setAttribute('tabindex', isActive ? '0' : '-1');
    }
  }
  this._onDimensionChange?.(level);
}
```
