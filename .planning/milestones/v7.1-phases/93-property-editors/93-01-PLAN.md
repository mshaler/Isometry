    export function coerceFieldValue(field: string, rawValue: unknown): CoercionResult

    Field classification constants (define inside the module):
    - NULLABLE_TEXT_FIELDS: `['content', 'summary', 'folder', 'status', 'url', 'mime_type', 'source', 'source_id', 'source_url', 'location_name']`

    Coercion logic:
    1. If field === 'name': reject empty/whitespace-only strings with `{ error: 'Name is required' }`. Otherwise return the trimmed string.

    7. If field === 'card_type': return rawValue (CardType string passthrough)
    8. If field === 'tags': return rawValue (string[] passthrough)
    9. Default: return rawValue (passthrough for unknown fields like id, created_at, etc.)
