<!-- SuperWidget DOM contract for selectors -->
DOM attributes used by SuperWidget:
- `[data-component="superwidget"]` — root element
- `[data-slot="header"]` — header slot
- `[data-slot="canvas"]` — canvas slot
- `[data-slot="tabs"]` — tabs slot
- `[data-slot="status"]` — status slot
- `[data-canvas-type="Explorer"|"View"|"Editor"]` — canvas stub type
- `[data-sidecar]` — sidecar element (View/Bound only)
- `[data-render-count]` — render count attribute

**2. Create `e2e/superwidget-smoke.spec.ts`:**

```typescript
import { test, expect } from '@playwright/test';

test.describe('SuperWidget WebKit smoke', () => {
  // Three-step transition matrix per D-08:
  // Explorer -> View/Bound (sidecar appears) -> Editor (zone label updates)

  test('transition matrix: Explorer -> View/Bound -> Editor', async ({ page }) => {
    // Navigate to the page with SuperWidget mounted
    await page.goto('/');  // or harness path

    // Step 1: Commit Explorer projection
    await page.evaluate(() => {
      // Access SuperWidget instance and commit Explorer projection
      // (implementation depends on harness approach)
    });

    // Assert Explorer canvas mounted
    await expect(page.locator('[data-canvas-type="Explorer"]')).toBeVisible();
    await expect(page.locator('[data-sidecar]')).toHaveCount(0);

    // Step 2: Commit View/Bound projection
    await page.evaluate(() => {
      // Commit View/Bound projection
    });

    // Assert View canvas with sidecar
    await expect(page.locator('[data-canvas-type="View"]')).toBeVisible();
    await expect(page.locator('[data-sidecar]')).toHaveCount(1);
    // Zone label updated
    await expect(page.locator('[data-slot="header"]')).toContainText('Secondary');

    // Step 3: Commit Editor projection
    await page.evaluate(() => {
      // Commit Editor projection
    });

    // Assert Editor canvas, no View trace
    await expect(page.locator('[data-canvas-type="Editor"]')).toBeVisible();
    await expect(page.locator('[data-canvas-type="View"]')).toHaveCount(0);
    await expect(page.locator('[data-slot="header"]')).toContainText('Tertiary');
  });
});
```

The spec should use `test.describe.configure({ retries: 0 })` for determinism. Keep it to ONE test with the three-step transition — fast, focused, proves real browser path.
