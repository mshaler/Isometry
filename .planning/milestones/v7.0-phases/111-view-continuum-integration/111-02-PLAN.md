```typescript
const handleKeyDown = useCallback((event: KeyboardEvent) => {
  // Only handle when Cmd (macOS) or Ctrl (Windows/Linux) is pressed
  if (!(event.metaKey || event.ctrlKey)) return;

  // Map number keys to modes
  const keyToMode: Record<string, GridContinuumMode> = {
    '1': 'gallery',
    '2': 'list',
    '3': 'kanban',
    '4': 'grid',
    '5': 'supergrid',
  };

  const mode = keyToMode[event.key];
  if (mode && mode !== currentMode) {
    event.preventDefault();
    onModeChange(mode);
  }
}, [currentMode, onModeChange]);
```

3. Register keyboard listener in useEffect:
```typescript
useEffect(() => {
  document.addEventListener('keydown', handleKeyDown);
  return () => document.removeEventListener('keydown', handleKeyDown);
}, [handleKeyDown]);
```
