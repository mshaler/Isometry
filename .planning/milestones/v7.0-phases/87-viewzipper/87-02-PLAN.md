```typescript
const mainEl = shell.getViewContentEl().parentElement!;
const viewContentEl = shell.getViewContentEl();

// Apply crossfade transition class to view content element (UI-SPEC: .vzip-transition-frame)
viewContentEl.classList.add('vzip-transition-frame');

viewZipper = new ViewZipper({
  container: mainEl,
  onSwitch: async (viewType: ViewType) => {
    // Crossfade: set opacity 0, switch view, then set opacity 1
    viewContentEl.style.opacity = '0';
    await viewManager.switchTo(viewType, () => viewFactory[viewType]());
    viewContentEl.style.opacity = '1';
  },
  announcer,
});
// Reposition: strip should be between panel-rail and view-content (VisualExplorer wraps view-content)
mainEl.insertBefore(viewZipper.getElement(), viewContentEl);
```

Note: The `.vzip-transition-frame` class (from `view-zipper.css`) applies `transition: opacity 300ms ease`, so setting `opacity = '0'` triggers a 300ms fade-out, then after `switchTo` resolves the view content is ready and `opacity = '1'` triggers a 300ms fade-in. This crossfade runs on every tab click and every auto-cycle step because both paths flow through the `onSwitch` callback.
