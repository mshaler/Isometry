**CatalogBridgeAdapter** — implements SuperGridBridgeLike by converting datasets:query results into CellDatum[]:

```typescript
class CatalogBridgeAdapter implements SuperGridBridgeLike {
  constructor(private _bridge: WorkerBridge) {}

  async superGridQuery(_config: SuperGridQueryConfig): Promise<CellDatum[]> {
    // Query datasets table via Plan 01 handler
    const datasets = await this._bridge.send('datasets:query', {});

    // Convert each dataset row into a CellDatum.
    // SuperGrid renders one row per dataset. Each visible field becomes a cell.
    // The col_key encodes the field name, row_key encodes the dataset id.
    // card_ids stores the dataset id for click identification.
    // card_names stores the display value for the cell.
    return datasets.flatMap((ds) => {
      const fields: Array<{ field: string; value: string }> = [
        { field: 'name', value: ds.name },
        { field: 'source_type', value: ds.source_type },
        { field: 'card_count', value: String(ds.card_count) },
        { field: 'connection_count', value: String(ds.connection_count) },
        { field: 'last_imported_at', value: new Date(ds.last_imported_at).toLocaleDateString() },
      ];
      return fields.map((f) => ({
        col_key: f.field,
        row_key: ds.id,
        count: ds.is_active ? 1 : 0,  // Encode is_active in count for row highlighting
        card_ids: [ds.id],
        card_names: [f.value],
      }));
    });
  }

  async calcQuery(): Promise<CalcQueryResult> {
    // No calc footer for catalog — return empty
    return { values: {} };
  }
}
```

NOTE: The exact CellDatum mapping above is the KEY design decision. SuperGrid expects cells with col_key/row_key/count/card_ids/card_names. By encoding each dataset field as a col_key and each dataset id as a row_key, SuperGrid renders a grid where each row is a dataset and each column is a field. The `count` field encodes `is_active` (1 = active, 0 = inactive) for row highlighting via the existing `.dexp-catalog-row--active` CSS class.

After reading SuperGrid.ts in detail, you may find that the CellDatum → rendering pipeline makes assumptions about the data shape (e.g., header generation from unique col_key values, row grouping). Adjust the CatalogBridgeAdapter mapping to match what _renderCells() actually expects. The critical constraint is: SuperGrid MUST render the datasets — do NOT fall back to a plain HTML table.

**CatalogSuperGrid class** — orchestrates mount/refresh/destroy:

```typescript
export interface CatalogSuperGridConfig {
  bridge: WorkerBridge;
  onDatasetClick: (datasetId: string, name: string, sourceType: string, isActive: boolean) => void;
}

export class CatalogSuperGrid {
  private _superGrid: SuperGrid | null = null;
  private _provider: CatalogProviderAdapter;
  private _bridgeAdapter: CatalogBridgeAdapter;
  private _filterAdapter: CatalogFilterAdapter;
  private _coordinatorCallbacks: Set<() => void> = new Set();
  private _config: CatalogSuperGridConfig;

  constructor(config: CatalogSuperGridConfig) {
    this._config = config;
    this._provider = new CatalogProviderAdapter();
    this._bridgeAdapter = new CatalogBridgeAdapter(config.bridge);
    this._filterAdapter = new CatalogFilterAdapter();
  }

  mount(container: HTMLElement): void {
    // Create a mini coordinator that CatalogSuperGrid controls
    const coordinator = {
      subscribe: (cb: () => void) => {
        this._coordinatorCallbacks.add(cb);
        return () => { this._coordinatorCallbacks.delete(cb); };
      },
    };

    this._superGrid = new SuperGrid(
      this._provider,
      this._filterAdapter,
      this._bridgeAdapter,
      coordinator,
    );
    this._superGrid.mount(container);
  }

  /** Trigger a re-fetch and re-render of the catalog grid */
  refresh(): void {
    for (const cb of this._coordinatorCallbacks) {
      cb();
    }
  }

  destroy(): void {
    this._superGrid?.destroy();
    this._superGrid = null;
    this._coordinatorCallbacks.clear();
  }
}
```

SuperGridSelect (selectionAdapter) has `select()` which is called on cell click — create a CatalogSelectionAdapter that intercepts select calls and routes to `onDatasetClick`. function showDataExplorer(): void {
  if (!dataExplorerMounted) {

    for (const child of Array.from(panelRailEl.children)) {
      (child as HTMLElement).style.display = 'none';

        void (async () => {
          const result = await bridge.send('etl:export', { format });
          const blob = new Blob([result.data], { type: 'text/plain;charset=utf-8' });
          const url = URL.createObjectURL(blob);
          const a = document.createElement('a');
          a.href = url;
          a.download = result.filename;
          a.click();
          URL.revokeObjectURL(url);
        })();
      },
      onExportDatabase: async () => {
        const data = await bridge.send('db:export', {});
        const blob = new Blob([data as Uint8Array], { type: 'application/x-sqlite3' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        const dateStr = new Date().toISOString().slice(0, 10);
        a.download = `isometry-backup-${dateStr}.sqlite`;
        a.click();
        URL.revokeObjectURL(url);
      },
      onVacuum: async () => {
        await bridge.send('datasets:vacuum', {});
      },
      onFileDrop: (file: File) => {

        const sourceMap: Record<string, string> = {
          json: 'json', csv: 'csv', xlsx: 'excel', xls: 'excel',
          md: 'markdown', html: 'html', htm: 'html',
        };
        const source = sourceMap[ext] ?? 'json';
        const binaryFormats = new Set(['xlsx', 'xls']);
        void (async () => {
          const data: string | ArrayBuffer = binaryFormats.has(ext)
            ? await file.arrayBuffer() : await file.text();
          await bridge.importFile(source as SourceType, data, { filename: file.name });
          coordinator.scheduleUpdate();
          void refreshDataExplorer();
        })();
      },

    dataExplorer.mount(panelRailEl);
    dataExplorerMounted = true;

    // Mount Catalog SuperGrid into the catalog body element
    const catalogBodyEl = dataExplorer.getCatalogBodyEl();

      catalogGrid = new CatalogSuperGrid({

              title: 'Switch Dataset?',

              cancelLabel: 'Keep Current Dataset',

              await handleDatasetSwitch(datasetId);
            }
          })();

  } else {
    // Already mounted — show it, hide workbench panels
    for (const child of Array.from(panelRailEl.children)) {
      const el = child as HTMLElement;
      if (el.classList.contains('data-explorer')) {
        el.style.display = '';
      } else {
        el.style.display = 'none';
