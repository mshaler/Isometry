1. Define hook interface:
```typescript

  enabled?: boolean;
  onTick?: (nodes: GraphNode[], links: GraphLink[]) => void;
  onEnd?: () => void;
}

export interface UseForceSimulationResult {
  reheat: (alpha?: number) => void;
  stop: () => void;
  state: SimulationState;
}
```

2. Implement hook with two useEffect patterns:

Effect 1 (manager creation - runs once):
```typescript
const managerRef = useRef<ForceSimulationManager | null>(null);
const [state, setState] = useState<SimulationState>('stopped');

useEffect(() => {
  managerRef.current = new ForceSimulationManager();
  return () => {
    managerRef.current?.destroy();
    managerRef.current = null;
  };
}, []);
```

Effect 2 (simulation lifecycle - runs on data change):
```typescript
useEffect(() => {
  if (!enabled || !containerRef.current || nodes.length === 0) {
    managerRef.current?.stop();
    setState('stopped');
    return;
  }

  managerRef.current?.start(containerRef.current, nodes, links, config, {
    onTick: (n, l) => {
      setState(managerRef.current?.getState() ?? 'stopped');
      onTick?.(n, l);
    },
    onEnd: () => {
      setState('stopped');
      onEnd?.();
    },
  });

  return () => {
    managerRef.current?.stop();
  };
}, [containerRef, nodes, links, config, enabled, onTick, onEnd]);
```

3. Return memoized callbacks:
```typescript
const reheat = useCallback((alpha?: number) => managerRef.current?.reheat(alpha), []);
const stop = useCallback(() => managerRef.current?.stop(), []);
return { reheat, stop, state };
```
