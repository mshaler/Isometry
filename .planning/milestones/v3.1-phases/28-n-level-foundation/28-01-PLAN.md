  <name>Task 2: Create shared compound key utility (keys.ts)</name>
  <files>src/views/supergrid/keys.ts, tests/views/supergrid/keys.test.ts</files>
  <behavior>
    - buildDimensionKey(cell, axes) → \x1f-joined string of axis values from cell
      - Single axis: buildDimensionKey({folder:'Work'}, [{field:'folder'}]) → 'Work'
      - Two axes: buildDimensionKey({folder:'Work', status:'Active'}, [{field:'folder'}, {field:'status'}]) → 'Work\x1fActive'
      - Three axes: buildDimensionKey({folder:'Work', status:'Active', priority:'1'}, [{field:'folder'}, {field:'status'}, {field:'priority'}]) → 'Work\x1fActive\x1f1'
      - Null/undefined values: buildDimensionKey({folder:null}, [{field:'folder'}]) → 'None'
      - Missing field: buildDimensionKey({}, [{field:'folder'}]) → 'None'
    - buildCellKey(cell, rowAxes, colAxes) → row dimension key + \x1e + col dimension key
      - Single axes: buildCellKey({folder:'Work', card_type:'note'}, [{field:'folder'}], [{field:'card_type'}]) → 'Work\x1enote'
      - Multi-level: 'Work\x1fActive\x1eNote\x1fHigh'
    - parseCellKey(cellKey) → { rowKey: string, colKey: string }
      - 'Work\x1enote' → { rowKey: 'Work', colKey: 'note' }
      - 'Work\x1fActive\x1eNote\x1fHigh' → { rowKey: 'Work\x1fActive', colKey: 'Note\x1fHigh' }
      - No \x1e separator returns { rowKey: cellKey, colKey: '' }
    - findCellInData(cellKey, cells, rowAxes, colAxes) → CellDatum | undefined
      - Reverse-lookup: parse key, match against cell axis fields
  </behavior>
