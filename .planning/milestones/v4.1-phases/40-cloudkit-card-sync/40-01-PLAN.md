5. Replace the existing `handleSentRecordZoneChanges` failure loop (currently just logs the error) with server-wins conflict resolution per CONTEXT.md locked decision:

```swift
for failedSave in sent.failedRecordSaves {
    let recordId = failedSave.record.recordID.recordName
    let error = failedSave.error

    if error.code == .serverRecordChanged,
       let serverRecord = error.serverRecord {
        // SYNC-04: Server wins silently -- accept server version
        logger.info("Conflict on \(recordId): accepting server version")

        // Archive server record's system fields (Pitfall 3 prevention)
        archiveSystemFields(for: serverRecord)

        // Forward resolved server record to JS SyncMerger
        var fields: [String: Any] = [:]
        if serverRecord.recordType == SyncConstants.cardRecordType {
            let codableFields = serverRecord.cardFieldsDictionary()
            for (key, value) in codableFields {
                switch value {
                case .string(let s): fields[key] = s
                case .int(let i): fields[key] = i
                case .double(let d): fields[key] = d
                case .bool(let b): fields[key] = b
                case .null: fields[key] = NSNull()
                }
            }
        }

        let recordDicts: [[String: Any]] = [[
            "recordType": serverRecord.recordType,
            "recordId": recordId,
            "operation": "save",
            "fields": fields,
        ]]
        let payload: [String: Any] = ["records": recordDicts]
        Task { @MainActor in
            self.bridgeManager?.sendSyncNotification(payload)
        }

        // Remove from pending changes (conflict resolved)
        let resolvedId = recordId
        pendingChanges.removeAll(where: { $0.recordId == resolvedId })
    } else {
        logger.error("Failed to send record \(recordId): \(error.localizedDescription)")
        // Update status to error
        let errorMsg = error.localizedDescription
        Task { @MainActor in
            self.statusPublisher?.status = .error(errorMsg)
        }
        // CKSyncEngine will retry automatically for transient errors
    }
}
```

**Create SyncStatusView.swift** (new file):

SwiftUI view for the sync status toolbar icon per CONTEXT.md locked decisions:
- 3 states: idle (cloud checkmark), syncing (animated cloud arrows), error (cloud with exclamation)
- Native SwiftUI toolbar icon, not JS-rendered
- Tapping error-state icon shows a brief popover with the error message
- No interrupting alerts or toasts

```swift
import SwiftUI

struct SyncStatusView: View {
    @ObservedObject var statusPublisher: SyncStatusPublisher
    @State private var showingErrorPopover = false

    var body: some View {
        Button {
            if case .error = statusPublisher.status {
                showingErrorPopover = true
            }
        } label: {
            Group {
                switch statusPublisher.status {
                case .idle:
                    Image(systemName: "checkmark.icloud")
                        .foregroundStyle(.secondary)
                case .syncing:
                    Image(systemName: "arrow.triangle.2.circlepath.icloud")
                        .foregroundStyle(.blue)
                        .symbolEffect(.rotate, options: .repeating)
                case .error:
                    Image(systemName: "exclamationmark.icloud")
                        .foregroundStyle(.red)
                }
            }
        }
        .buttonStyle(.plain)
        .popover(isPresented: $showingErrorPopover) {
            if case .error(let message) = statusPublisher.status {
                Text(message)
                    .font(.caption)
                    .padding()
                    .frame(maxWidth: 250)
            }
        }
    }
}
```

Note on `.symbolEffect(.rotate)`: This requires iOS 17+ / macOS 14+ which matches the deployment target. If `.rotate` is not available for the chosen SF Symbol, use `.pulse` or a simple opacity animation instead. Check what compiles.
