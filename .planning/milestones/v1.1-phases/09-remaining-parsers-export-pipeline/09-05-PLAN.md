```typescript
private parsers = {
  apple_notes: new AppleNotesParser(),
  markdown: new MarkdownParser(),
  csv: new CSVParser(),
  json: new JSONParser(),
  excel: new ExcelParser(),
  html: new HTMLParser(),
};

async import(request: ImportRequest): Promise<ImportResult> {
  // ... existing setup

  // Parse based on source type
  let parseResult: ParseResult;

  switch (request.source) {
    case 'apple_notes': {
      const files = this.prepareAppleNotesFiles(request.data);
      parseResult = this.parsers.apple_notes.parse(files);
      break;
    }
    case 'markdown': {
      const files = this.prepareMarkdownFiles(request.data);
      parseResult = this.parsers.markdown.parse(files, request.options);
      break;
    }
    case 'csv': {
      parseResult = this.parsers.csv.parse(request.data as string, request.options);
      break;
    }
    case 'json': {
      parseResult = this.parsers.json.parse(request.data as string, request.options);
      break;
    }
    case 'excel': {
      parseResult = await this.parsers.excel.parse(request.data as ArrayBuffer, request.options);
      break;
    }
    case 'html': {
      const htmlStrings = Array.isArray(request.data) ? request.data : [request.data];
      parseResult = this.parsers.html.parse(htmlStrings, request.options);
      break;
    }
    default: {
      const _exhaustive: never = request.source;
      throw new Error(`Unknown source type: ${_exhaustive}`);
    }
  }

  // ... rest of pipeline (dedup -> write -> catalog)
}
```

```typescript
import type { Database } from 'sql.js';
import type { ETLExportRequest, ETLExportResponse } from '../protocol';
import { ExportOrchestrator } from '../../etl/ExportOrchestrator';

export async function handleETLExport(
  db: Database,
  payload: ETLExportRequest['payload']
): Promise<ETLExportResponse['result']> {
  const orchestrator = new ExportOrchestrator(db);

  const result = orchestrator.export(payload.format, {
    cardIds: payload.cardIds,
    excludeDeleted: true,
  });

  return {
    data: result.data,
    filename: result.filename,
  };
}
```

```typescript
import { describe, it, expect, beforeEach } from 'vitest';
import initSqlJs, { Database } from 'sql.js';
import { ImportOrchestrator } from '../../src/etl/ImportOrchestrator';
import { ExportOrchestrator } from '../../src/etl/ExportOrchestrator';
import { MarkdownParser } from '../../src/etl/parsers/MarkdownParser';
import { loadSchema } from '../../src/database/schema';

describe('ETL Export Round-Trip', () => {
  let db: Database;

  beforeEach(async () => {
    const SQL = await initSqlJs();
    db = new SQL.Database();
    const schema = await loadSchema();
    db.run(schema);

    // Seed with test data
    db.run(`
      INSERT INTO cards (id, card_type, name, content, tags, source, source_id, created_at, modified_at)
      VALUES
        ('card-1', 'note', 'Note One', 'Content one', '["tag1"]', 'test', '1', '2024-01-01T00:00:00Z', '2024-01-01T00:00:00Z'),
        ('card-2', 'note', 'Note Two', 'Content two', '["tag2"]', 'test', '2', '2024-01-02T00:00:00Z', '2024-01-02T00:00:00Z')
    `);
  });

  it('markdown export round-trips through parser', async () => {
    const exportOrchestrator = new ExportOrchestrator(db);
    const result = exportOrchestrator.export('markdown', {});

    // Parse exported markdown
    const parser = new MarkdownParser();
    // Split exported file back into individual files
    const files = result.data.split('\n\n---\n\n').map((content, i) => ({
      path: `note-${i}.md`,
      content,
    }));

    const parseResult = parser.parse(files);

    expect(parseResult.cards.length).toBe(2);
    expect(parseResult.cards.map(c => c.name).sort()).toEqual(['Note One', 'Note Two']);
  });

  it('export filters by cardIds', async () => {
    const orchestrator = new ExportOrchestrator(db);
    const result = orchestrator.export('json', { cardIds: ['card-1'] });

    const data = JSON.parse(result.data);
    expect(data.cards.length).toBe(1);
    expect(data.cards[0].name).toBe('Note One');
  });

  it('export excludes deleted cards', async () => {
    db.run("UPDATE cards SET deleted_at = '2024-01-15T00:00:00Z' WHERE id = 'card-1'");

    const orchestrator = new ExportOrchestrator(db);
    const result = orchestrator.export('json', {});

    const data = JSON.parse(result.data);
    expect(data.cards.length).toBe(1);
    expect(data.cards[0].name).toBe('Note Two');
  });
});
```
