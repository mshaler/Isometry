```typescript
export class ExcelParser {
  private xlsx: typeof import('xlsx') | null = null;

  async parse(buffer: ArrayBuffer, options?: ExcelParseOptions): Promise<ParseResult> {
    // Dynamic import on first use (P27 - bundle size)
    if (!this.xlsx) {
      this.xlsx = await import('xlsx');
    }

    const workbook = this.xlsx.read(buffer, {
      type: 'array',
      cellDates: true, // Convert serial numbers to Date objects
    });

    const sheetName = options?.sheet ?? workbook.SheetNames[0];
    const sheet = workbook.Sheets[sheetName];
    const rows = this.xlsx.utils.sheet_to_json(sheet);

    // Process rows into CanonicalCards...
  }
}
```

**Key implementation details per RESEARCH.md:**
- Dynamic import: `await import('xlsx')` not top-level import
- cellDates: true required to avoid serial number confusion
- Date handling: `value instanceof Date ? value.toISOString() : ...`
- Excel sheets: Import first/active sheet only (per CONTEXT.md)
