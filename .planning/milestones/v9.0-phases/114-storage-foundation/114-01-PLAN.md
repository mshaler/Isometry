Create `src/worker/utils/sanitize.ts` exporting:
```typescript
export function sanitizeAlgorithmResult(row: Record<string, unknown>): Record<string, unknown> {
  const METRIC_FIELDS = ['centrality', 'pagerank', 'community_id', 'clustering_coeff', 'sp_depth', 'in_spanning_tree'] as const;
  const sanitized = { ...row };
  for (const field of METRIC_FIELDS) {
    const val = sanitized[field];
    if (typeof val === 'number' && (!Number.isFinite(val))) {
      sanitized[field] = null;
    }
  }
  return sanitized;
}
```

This converts NaN and Infinity to null for all 6 algorithm output fields. Uses `Number.isFinite()` which returns false for NaN, Infinity, and -Infinity.
