         ```
         const lastMutation = manager.getHistory()[manager.getHistory().length - 1];

         const didUndo = await manager.undo();
         if (didUndo && toast) {
           toast.show(`Undid: ${description}`);
         }
         ```
         NOTE: Must capture description BEFORE calling undo() since undo() pops from history.

         }
