```typescript
const wrapper = root.querySelector('.pv-grid-wrapper') as HTMLElement | null;
```
This returns null in the PivotGrid DOM. Instead, apply density classes directly to `root`:

```typescript
function _applyDensityClass(el: HTMLElement | null): void {
  if (!el) return;
  for (const level of DENSITY_LEVELS) {
    el.classList.remove(`pv-density--${level}`);
  }
  if (densityState.level !== 'normal') {
    el.classList.add(`pv-density--${densityState.level}`);
  }
}
```

2. In `afterRender`, pass `root` directly to `_applyDensityClass(root)` instead of querying a child. Also store `root` in a module-level variable so the click handler can use it:
