<objective>
Implement graph traversal queries using TDD with recursive CTEs, supporting depth-limited breadth traversal and shortest path finding.

Purpose: Graph traversal is what makes Isometry's connection model useful. Users need to discover related cards (e.g., "show me everything connected to this card within 3 hops"). The recursive CTE approach runs entirely within SQLite/WASM, using the connection indexes from Phase 1, instead of pulling all edges to JavaScript for BFS/DFS. This plan establishes the query patterns that Phase 5's network and tree views will consume.

Output: Working graph traversal functions with cycle prevention, depth limiting, and bidirectional edge following.
</objective>

```typescript
import type { Database } from '../Database';
import type { CardWithDepth } from './types';
import { rowToCard } from './helpers';
import type { SqlValue } from 'sql.js';

// Graph traversal using recursive CTE with UNION for cycle prevention
export function connectedCards(
  db: Database,
  startId: string,
  maxDepth: number = 3
): CardWithDepth[] {
  if (maxDepth <= 0) return [];

  // UNION (not UNION ALL) prevents revisiting the same node (cycle prevention)
  // Follows connections bidirectionally: source_id OR target_id
  const result = db.exec(
    `WITH RECURSIVE traversal(card_id, depth) AS (
       -- Base case: start node at depth 0
       SELECT ?, 0
       UNION
       -- Recursive case: follow connections in both directions
       SELECT
         CASE
           WHEN c.source_id = t.card_id THEN c.target_id
           ELSE c.source_id
         END,
         t.depth + 1
       FROM traversal t
       JOIN connections c ON c.source_id = t.card_id OR c.target_id = t.card_id
       WHERE t.depth < ?
     )
     SELECT DISTINCT cards.*, traversal.depth
     FROM traversal
     JOIN cards ON cards.id = traversal.card_id
     WHERE traversal.card_id != ?
       AND cards.deleted_at IS NULL
     ORDER BY traversal.depth, cards.name`,
    [startId, maxDepth, startId]
  );

  if (!result[0]) return [];
  const { columns, values } = result[0];
  return values.map(row => {
    const obj: Record<string, unknown> = {};
    columns.forEach((col, i) => { obj[col] = row[i]; });
    const depth = obj['depth'] as number;
    return { card: rowToCard(obj as Record<string, SqlValue>), depth };
  });
}

// Shortest path between two cards using recursive CTE with path tracking
export function shortestPath(
  db: Database,
  fromId: string,
  toId: string
): string[] | null {
  if (fromId === toId) return [];

  // BFS via recursive CTE with path accumulation
  // Path is stored as concatenated IDs with separator for SQLite TEXT handling
  const result = db.exec(
    `WITH RECURSIVE pathfinder(card_id, path, depth) AS (
       SELECT ?, ?, 0
       UNION
       SELECT
         CASE
           WHEN c.source_id = p.card_id THEN c.target_id
           ELSE c.source_id
         END,
         p.path || ',' ||
         CASE
           WHEN c.source_id = p.card_id THEN c.target_id
           ELSE c.source_id
         END,
         p.depth + 1
       FROM pathfinder p
       JOIN connections c ON c.source_id = p.card_id OR c.target_id = p.card_id
       WHERE p.depth < 10
         AND p.path NOT LIKE '%' ||
           CASE
             WHEN c.source_id = p.card_id THEN c.target_id
             ELSE c.source_id
           END || '%'
     )
     SELECT path FROM pathfinder
     WHERE card_id = ?
     ORDER BY depth
     LIMIT 1`,
    [fromId, fromId, toId]
  );

  if (!result[0]?.values[0]) return null;
  const pathStr = result[0].values[0][0] as string;
  return pathStr.split(',');
}
```

**Key implementation notes:**
- `UNION` (not `UNION ALL`) in connectedCards for automatic cycle prevention per Pitfall 5 from research
- Bidirectional edge following via `CASE WHEN c.source_id = t.card_id THEN c.target_id ELSE c.source_id END`
- `WHERE traversal.card_id != ?` excludes the start card from results
- `AND cards.deleted_at IS NULL` excludes soft-deleted cards
- `ORDER BY traversal.depth, cards.name` gives deterministic ordering for tests
- shortestPath uses path string accumulation with LIKE check for visited-node prevention (alternative to UNION dedup since we need the full path)
- shortestPath hard-limits to depth 10 to prevent unbounded recursion on large graphs

**Performance note:** At 10K cards / 50K connections, depth-3 traversal uses `idx_conn_source` and `idx_conn_target` indexes. PERF-04 threshold (<500ms p95) is validated in Plan 05's benchmark suite, not here. This plan verifies functional correctness on small test graphs.
