```typescript
import type { Database } from '../../src/database/Database';

export const SEED_CONFIG = {
  cardCount: 10_000,
  connectionCount: 50_000,
  avgContentLength: 500,
} as const;

// Word lists for generating realistic content
const WORDS = [
  'knowledge', 'management', 'system', 'project', 'planning', 'data',
  'analysis', 'design', 'implementation', 'testing', 'deployment',
  'architecture', 'interface', 'database', 'query', 'performance',
  'optimization', 'algorithm', 'structure', 'pattern', 'framework',
  'component', 'module', 'service', 'layer', 'abstraction',
  // ... ~100 varied words for realistic FTS content
];

const CARD_TYPES = ['note', 'task', 'event', 'resource', 'person'] as const;
const FOLDERS = ['work', 'personal', 'research', 'archive', null];
const STATUSES = ['active', 'pending', 'completed', 'archived', null];
const LABELS = ['mentions', 'contains', 'references', 'depends_on', 'related_to'];

function randomContent(avgLength: number): string {
  // Generate random text of approximately avgLength characters
  // using WORDS array for FTS-searchable content
}

function randomElement<T>(arr: readonly T[]): T {
  return arr[Math.floor(Math.random() * arr.length)]!;
}

export interface SeedResult {
  hubCardId: string;  // A well-connected card for graph benchmarks
  sampleCardIds: string[];  // First 100 card IDs for reference
}

export function seedDatabase(db: Database): SeedResult {
  const cardIds: string[] = [];
  const now = new Date().toISOString();

  // Seed cards in a single transaction
  db.run('BEGIN');
  for (let i = 0; i < SEED_CONFIG.cardCount; i++) {
    const id = crypto.randomUUID();
    cardIds.push(id);
    const cardType = randomElement(CARD_TYPES);
    const folder = randomElement(FOLDERS);
    const status = randomElement(STATUSES);
    const tags = JSON.stringify(
      Array.from({ length: Math.floor(Math.random() * 4) }, () => randomElement(WORDS))
    );
    db.run(
      `INSERT INTO cards(id, card_type, name, content, folder, tags, status, priority, created_at, modified_at)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
      [
        id,
        cardType,
        `Card ${i}: ${randomElement(WORDS)} ${randomElement(WORDS)}`,
        randomContent(SEED_CONFIG.avgContentLength),
        folder,
        tags,
        status,
        Math.floor(Math.random() * 5),
        now,
        now,
      ]
    );
  }
  db.run('COMMIT');

  // Seed connections in a single transaction
  // Create a "hub" card with many connections for graph traversal benchmarks
  const hubCardId = cardIds[0]!;

  db.run('BEGIN');
  // First 200 connections: hub card to random cards (ensures hub is well-connected)
  for (let i = 0; i < 200 && i < SEED_CONFIG.connectionCount; i++) {
    const targetIdx = 1 + Math.floor(Math.random() * (cardIds.length - 1));
    try {
      db.run(
        `INSERT OR IGNORE INTO connections(id, source_id, target_id, label, weight)
         VALUES (?, ?, ?, ?, ?)`,
        [
          crypto.randomUUID(),
          hubCardId,
          cardIds[targetIdx]!,
          randomElement(LABELS),
          Math.random(),
        ]
      );
    } catch { /* skip duplicates */ }
  }

  // Remaining connections: random pairs
  for (let i = 200; i < SEED_CONFIG.connectionCount; i++) {
    const srcIdx = Math.floor(Math.random() * cardIds.length);
    let tgtIdx = Math.floor(Math.random() * cardIds.length);
    if (tgtIdx === srcIdx) tgtIdx = (tgtIdx + 1) % cardIds.length;
    try {
      db.run(
        `INSERT OR IGNORE INTO connections(id, source_id, target_id, label, weight)
         VALUES (?, ?, ?, ?, ?)`,
        [
          crypto.randomUUID(),
          cardIds[srcIdx]!,
          cardIds[tgtIdx]!,
          randomElement(LABELS),
          Math.random(),
        ]
      );
    } catch { /* skip duplicates */ }
  }
  db.run('COMMIT');

  return {
    hubCardId,
    sampleCardIds: cardIds.slice(0, 100),
  };
}
```

**Key design decisions:**
- `INSERT OR IGNORE` for connections to handle UNIQUE constraint collisions on random pairs
- Hub card (cardIds[0]) gets 200 direct connections, ensuring rich graph for PERF-04
- Single BEGIN/COMMIT for cards, single BEGIN/COMMIT for connections (fast seeding per Pitfall 6)
- Word list provides FTS-searchable vocabulary for PERF-03
- Returns hubCardId for graph traversal benchmark target

```typescript
import { bench, describe, beforeAll, afterAll } from 'vitest';
import { Database } from '../../src/database/Database';
import { createCard } from '../../src/database/queries/cards';
import { searchCards } from '../../src/database/queries/search';
import { connectedCards } from '../../src/database/queries/graph';
import { seedDatabase, type SeedResult } from './seed';

// Module-level setup: seed once for all benchmarks
let db: Database;
let seed: SeedResult;

// Note on lifecycle: Vitest bench() supports beforeAll/afterAll in describe blocks.
// The database is seeded once and reused across all benchmarks.

describe('Performance Benchmarks (PERF-01..04)', () => {
  beforeAll(async () => {
    db = new Database();
    await db.initialize();
    seed = seedDatabase(db);
  });

  afterAll(() => {
    db.close();
  });

  // =====================================================================
  // PERF-01: Single card insert p95 <10ms
  // Using p99 as conservative proxy (tinybench does not expose p95)
  // If p99 < 10ms, then p95 necessarily < 10ms
  // =====================================================================
  bench('PERF-01: single card insert (<10ms p95)', () => {
    createCard(db, {
      name: `Bench Card ${Math.random()}`,
      content: 'Performance test card with some content for benchmarking insert speed',
      card_type: 'note',
      folder: 'benchmarks',
    });
  }, {
    iterations: 100,
    time: 5000,  // 5 second time limit
  });

  // =====================================================================
  // PERF-02: Bulk 1000-card insert p95 <1s
  // Each iteration inserts 1000 cards in a single transaction
  // =====================================================================
  bench('PERF-02: bulk 1000-card insert (<1s p95)', () => {
    const now = new Date().toISOString();
    db.run('BEGIN');
    for (let i = 0; i < 1000; i++) {
      db.run(
        `INSERT INTO cards(id, card_type, name, content, created_at, modified_at)
         VALUES (?, ?, ?, ?, ?, ?)`,
        [
          crypto.randomUUID(),
          'note',
          `Bulk ${i} ${Math.random()}`,
          'Bulk insert benchmark card with enough content to be realistic for FTS indexing',
          now,
          now,
        ]
      );
    }
    db.run('COMMIT');
  }, {
    iterations: 10,
    time: 30000,  // 30 second time limit
  });

  // =====================================================================
  // PERF-03: FTS search p95 <100ms on 10K cards
  // 3-word query per REQUIREMENTS.md specification
  // =====================================================================
  bench('PERF-03: FTS search 10K cards (<100ms p95)', () => {
    searchCards(db, 'knowledge management system', 20);
  }, {
    iterations: 100,
    time: 15000,  // 15 second time limit
  });

  // =====================================================================
  // PERF-04: Graph traversal depth 3 p95 <500ms
  // Uses the hub card (well-connected) from seed
  // =====================================================================
  bench('PERF-04: graph traversal depth 3 (<500ms p95)', () => {
    connectedCards(db, seed.hubCardId, 3);
  }, {
    iterations: 20,
    time: 30000,  // 30 second time limit
  });
});
```

**Running benchmarks:**
Vitest bench files use `.bench.ts` suffix. Run with:
```bash
npx vitest bench tests/database/performance.bench.ts
```

**p95 vs p99 documentation:**
tinybench (Vitest's bench engine) exposes `p75`, `p99`, `p995`, `p999` but NOT `p95`. Per research Open Question 1, we use p99 as a conservative proxy: if p99 < threshold, then p95 necessarily passes (p95 <= p99 by definition). The benchmark table output will show p99 values. If p99 is close to but above threshold, fall back to manual p95 calculation from samples.

**Threshold assertions:**
Vitest bench() does not natively assert on percentile values — it reports them. To enforce thresholds, create a companion test file `tests/database/performance-assertions.test.ts` that runs the same operations with `performance.now()` timing and asserts p95 < threshold. This is the fallback approach from research Open Question 1.

```typescript
import { describe, it, expect, beforeAll, afterAll } from 'vitest';

function computeP95(samples: number[]): number {
  const sorted = [...samples].sort((a, b) => a - b);
  const idx = Math.ceil(0.95 * sorted.length) - 1;
  return sorted[Math.max(0, idx)] ?? 0;
}

describe('Performance Threshold Assertions', () => {
```
