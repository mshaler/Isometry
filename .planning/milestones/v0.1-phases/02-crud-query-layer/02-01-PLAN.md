// For one-shot queries, prefer db.exec() or db.run() (no manual free needed).
export function withStatement<T>(

  sql: string,
  fn: (stmt: /* sql.js Statement */ any) => T

  // NOTE: Database.ts does not expose prepare() yet.

// Map sql.js row object to typed Connection
export function rowToConnection(row: Record<string, SqlValue>): Connection {
  return {
    id: row['id'] as string,
    source_id: row['source_id'] as string,
    target_id: row['target_id'] as string,

// Convert db.exec() output to Card array (zips columns with values)
export function execRowsToCards(result: { columns: string[]; values: unknown[][] }[]): Card[] {
  if (!result[0]) return [];
  const { columns, values } = result[0];
  return values.map(row => {
    const obj: Record<string, unknown> = {};
    columns.forEach((col, i) => { obj[col] = row[i]; });
    return rowToCard(obj as Record<string, SqlValue>);
  });
}

// Convert db.exec() output to Connection array
export function execRowsToConnections(result: { columns: string[]; values: unknown[][] }[]): Connection[] {
  if (!result[0]) return [];
  const { columns, values } = result[0];
  return values.map(row => {
    const obj: Record<string, unknown> = {};
    columns.forEach((col, i) => { obj[col] = row[i]; });
    return rowToConnection(obj as Record<string, SqlValue>);
  });
}

**CRITICAL: withStatement and Database.prepare().**
The current Database.ts only exposes `exec()` and `run()`, not `prepare()`. The research recommends using `db.exec()` and `db.run()` for one-shot queries (no prepare/free needed). The `withStatement` helper is defined as the PATTERN but may not be actively used until Phase 3+ where performance-critical prepared statement reuse is needed. For Phase 2:
- Use `db.run()` for INSERT/UPDATE/DELETE (mutations)
- Use `db.exec()` for SELECT (returns all rows)
- Do NOT add `prepare()` to Database.ts yet — defer to performance optimization if benchmarks require it

If the executor determines that `Database.prepare()` is needed for benchmarks, they should add it to Database.ts at that point and wire up `withStatement`. The helper exists as a documented pattern.
