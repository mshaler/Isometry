**Describe block 1: Type correctness (11 test cases)**

```typescript
test.describe('Type correctness: 11 alto-index subdirectory types', () => {
  test.beforeAll(async ({ browser }) => {
    // Import all alto-index fixtures once for the entire block
    const page = await browser.newPage();
    await waitForAppReady(page);
    await resetDatabase(page);
    await importAltoIndex(page);
    await page.close();
  });

  // NOTE: Since beforeAll imports data and tests are sequential,
  // each test just queries db state. But Playwright's beforeAll
  // gives each test its own page, so we need to re-navigate.

  for (const typeInfo of ALTO_TYPES) {
    test(`${typeInfo.file} imports as card_type='${typeInfo.expectedCardType}' with source='${typeInfo.source}'`, async ({ page }) => {
      await waitForAppReady(page);

      // Query cards with this source
      const result = await page.evaluate(async (source) => {
        const { queryAll } = (window as any).__isometry;
        return queryAll(
          'SELECT card_type, COUNT(*) as cnt FROM cards WHERE source = ? AND deleted_at IS NULL GROUP BY card_type',
          [source]
        );
      }, typeInfo.source);

      const rows = (result as { rows: Array<{ card_type: string; cnt: number }> }).rows;
      expect(rows.length, `Expected rows for source ${typeInfo.source}`).toBeGreaterThanOrEqual(1);

      // All cards for this source should have the expected card_type
      const matchingRow = rows.find(r => r.card_type === typeInfo.expectedCardType);
      expect(matchingRow, `Expected card_type='${typeInfo.expectedCardType}' for source='${typeInfo.source}'`).toBeTruthy();
      expect(matchingRow!.cnt).toBeGreaterThanOrEqual(1);
    });
  }
});
```

test.describe('Purge-then-replace behavior', () => {
  test('alto-index import purges pre-existing cards and connections', async ({ page }) => {
    await waitForAppReady(page);
    await resetDatabase(page);

    // Step 1: Seed with non-alto-index data (sample data)
    await page.evaluate(async () => {
      const iso = (window as any).__isometry;
      await iso.sampleManager.load('meryl-streep');
      iso.coordinator.scheduleUpdate();
    });
    await page.waitForTimeout(500);

    const seedCount = await getCardCount(page);
    expect(seedCount, 'Sample data should produce cards').toBeGreaterThan(40);

    // Verify connections exist from sample data
    const seedConnections = await page.evaluate(async () => {
      const { queryAll } = (window as any).__isometry;
      return queryAll('SELECT COUNT(*) as cnt FROM connections', []);
    });
    const seedConnCount = ((seedConnections as { rows: Array<{ cnt: number }> }).rows[0]?.cnt) ?? 0;
    expect(seedConnCount, 'Sample data should produce connections').toBeGreaterThan(0);

    // Step 2: Import alto-index (triggers purge-then-replace)
    await importAltoIndex(page);

    // Step 3: Assert sample cards are gone
    const postImportSampleCards = await page.evaluate(async () => {
      const { queryAll } = (window as any).__isometry;
      // Sample data uses source='sample' or similar non-alto source
      return queryAll(
        "SELECT COUNT(*) as cnt FROM cards WHERE source NOT LIKE 'alto_%' AND deleted_at IS NULL",
        []
      );
    });
    const remainingNonAlto = ((postImportSampleCards as { rows: Array<{ cnt: number }> }).rows[0]?.cnt) ?? 0;
    expect(remainingNonAlto, 'No non-alto cards should remain after purge-then-replace').toBe(0);

    // Step 4: Assert connections table is also purged of old data
    // (alto-index cards have no connections, so connections should be 0 or only alto-generated)
    const postImportConnections = await page.evaluate(async () => {
      const { queryAll } = (window as any).__isometry;
      return queryAll('SELECT COUNT(*) as cnt FROM connections', []);
    });
    const postConnCount = ((postImportConnections as { rows: Array<{ cnt: number }> }).rows[0]?.cnt) ?? 0;
    // Connections from sample data should be gone (purge deletes ALL connections)
    expect(postConnCount, 'Connections should be purged (all old connections deleted)').toBe(0);

    // Step 5: Assert alto-index cards are present
    const altoCardCount = await getCardCount(page);
    expect(altoCardCount, 'Alto-index cards should be present after purge+import').toBeGreaterThan(490);
  });
});
