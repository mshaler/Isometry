class PerformanceRegressionDetector {

    func detectRegressions() async -> RegressionDetectionResult {
        let currentPerformance = await measureCurrentPerformance()
        let baselinePerformance = await loadBaselinePerformance()

        let regressions = analyzePerformanceRegressions(
            current: currentPerformance,
            baseline: baselinePerformance,
            thresholds: RegressionThresholds.default
