    4. Add keyboard navigation to `TabBar`:

    Wire a `keydown` event listener on `_stripEl` (event delegation):
    ```typescript
    this._stripEl.addEventListener('keydown', (e: KeyboardEvent) => {
      const tabs = this._config.tabs;
      const currentIndex = tabs.findIndex(t => t.tabId === this._config.activeTabId);
      let nextIndex: number | null = null;

      if (e.key === 'ArrowRight') nextIndex = (currentIndex + 1) % tabs.length;
      else if (e.key === 'ArrowLeft') nextIndex = (currentIndex - 1 + tabs.length) % tabs.length;
      else if (e.key === 'Home') nextIndex = 0;
      else if (e.key === 'End') nextIndex = tabs.length - 1;

      if (nextIndex !== null && nextIndex !== currentIndex) {
        e.preventDefault();
        this._config.onSwitch(tabs[nextIndex]!.tabId);
        // Focus will be set by the update() cycle that onSwitch triggers
      }
    });
    ```
