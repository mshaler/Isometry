3. Create cell click handler that syncs to SelectionContext:
   ```typescript
   const handleCellClick = useCallback((
     cell: DataCell | undefined,
     rowPath: string[],
     colPath: string[]
   ) => {
     if (!cell) {
       // Clicked empty cell - clear selection
       setSelectedIds([]);
       contextLogger.debug('[IntegratedLayout] Empty cell clicked, selection cleared');
       return;
     }

     // Extract node ID from rawValue
     const nodeData = cell.rawValue as { id: string } | undefined;
     if (!nodeData?.id) return;

     // Update global selection state
     setSelectedIds([nodeData.id]);

     contextLogger.debug('[IntegratedLayout] Cell selected', {
       nodeId: nodeData.id,
       rowPath,
       colPath
     });
   }, [setSelectedIds]);
   ```

4. Pass handler to SuperGridCSS:
   ```typescript
   <SuperGridCSS
     rowAxis={rowAxis}
     columnAxis={colAxis}
     data={dataCells}
     theme={isNeXTSTEP ? 'nextstep' : 'modern'}
     onCellClick={handleCellClick}
     onHeaderClick={(type, path) => {
       contextLogger.debug('[IntegratedLayout] Header clicked', { type, path });
     }}
   />
   ```

5. Add selection change callback (optional, for future multi-select):
   ```typescript
   onSelectionChange={(selected) => {
     if (!selected) {
       setSelectedIds([]);
     }
     // Note: Full multi-select via row/col selection is Phase 107+ feature
   }}
   ```
