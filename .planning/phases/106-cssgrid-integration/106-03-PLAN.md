   ```typescript
         id: m.facet,
         name: m.facet,
         axis: m.axis as 'L' | 'A' | 'T' | 'C' | 'H',
         sourceColumn: m.facet,
         dataType: 'text' as const,
         sortOrder: 'asc' as const
       }));
   }, [pafvState.mappings]);

   const colFacets = useMemo(() => {
     return pafvState.mappings
       .filter(m => m.plane === 'x')
       .map(m => ({
         id: m.facet,
         name: m.facet,
         axis: m.axis as 'L' | 'A' | 'T' | 'C' | 'H',
         sourceColumn: m.facet,
         dataType: 'text' as const,
         sortOrder: 'asc' as const
       }));
   }, [pafvState.mappings]);
   ```

   ```typescript
     whereClause: buildWhereClause().clause || undefined
   ```
