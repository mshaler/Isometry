```typescript
// 11b. Data Explorer panel — mounts into panel rail when activated from SidebarNav
let dataExplorer: DataExplorerPanel | null = null;
let dataExplorerMounted = false;

// Helper: toggle panel rail between workbench panels and Data Explorer
const panelRailEl = document.querySelector('.workbench-panel-rail') as HTMLElement;

function showDataExplorer(): void {
  if (!dataExplorerMounted) {
    // Hide all existing workbench CollapsibleSections
    for (const child of Array.from(panelRailEl.children)) {
      if (child !== dataExplorer?._rootEl) {
        (child as HTMLElement).style.display = 'none';
      }
    }
    panelRailEl.setAttribute('data-active-panel', 'data-explorer');

    dataExplorer = new DataExplorerPanel({
      onImportFile: importFileHandler,
      onExport: (format) => {
        const result = new ExportOrchestrator(/* need db reference */).export(format);
        // Trigger download
        const blob = new Blob([result.data], { type: 'text/plain;charset=utf-8' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = result.filename;
        a.click();
        URL.revokeObjectURL(url);
      },
      onExportDatabase: async () => {
        const data = await bridge.send('db:export', {});
        const blob = new Blob([data as Uint8Array], { type: 'application/x-sqlite3' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        const dateStr = new Date().toISOString().slice(0, 10);
        a.download = `isometry-backup-${dateStr}.sqlite`;
        a.click();
        URL.revokeObjectURL(url);
      },
      onVacuum: async () => {
        await bridge.send('datasets:vacuum', {});
      },
      onFileDrop: (file: File) => {
        // Reuse existing file import logic from importFileHandler
        const ext = file.name.split('.').pop()?.toLowerCase() ?? '';
        const sourceMap: Record<string, string> = {
          json: 'json', csv: 'csv', xlsx: 'excel', xls: 'excel',
          md: 'markdown', html: 'html', htm: 'html',
        };
        const source = sourceMap[ext] ?? 'json';
        const binaryFormats = new Set(['xlsx', 'xls']);
        void (async () => {
          const data: string | ArrayBuffer = binaryFormats.has(ext)
            ? await file.arrayBuffer() : await file.text();
          await bridge.importFile(source as SourceType, data, { filename: file.name });
          coordinator.scheduleUpdate();
          // Refresh catalog and stats after import
          void refreshDataExplorer();
        })();
      },
    });
    dataExplorer.mount(panelRailEl);
    dataExplorerMounted = true;

    // Fetch and display stats + catalog data
    void refreshDataExplorer();
  } else {
    // Already mounted — just show it
    for (const child of Array.from(panelRailEl.children)) {
      const el = child as HTMLElement;
      if (el.classList.contains('data-explorer')) {
        el.style.display = '';
      } else {
        el.style.display = 'none';
      }
    }
    panelRailEl.setAttribute('data-active-panel', 'data-explorer');
    void refreshDataExplorer();
  }
}

function hideDataExplorer(): void {
  if (dataExplorerMounted && dataExplorer) {
    // Hide Data Explorer root
    const rootEl = panelRailEl.querySelector('.data-explorer') as HTMLElement | null;
    if (rootEl) rootEl.style.display = 'none';
    // Show all workbench sections
    for (const child of Array.from(panelRailEl.children)) {
      const el = child as HTMLElement;
      if (!el.classList.contains('data-explorer')) {
        el.style.display = '';
      }
    }
    panelRailEl.removeAttribute('data-active-panel');
  }
}
```

**3. Refresh helper for catalog and stats:**

```typescript
async function refreshDataExplorer(): Promise<void> {
  if (!dataExplorer) return;

  // Fetch DB stats
  const stats = await bridge.send('datasets:stats', {}) as { card_count: number; connection_count: number; db_size_bytes: number };
  dataExplorer.updateStats(stats);

  // Fetch datasets for catalog rendering
  const datasets = await bridge.send('datasets:query', {}) as Array<{
    id: string; name: string; source_type: string; card_count: number;
    connection_count: number; is_active: number; last_imported_at: string;
  }>;

  // Render catalog as a simple HTML table (NOT full SuperGrid)
  // Rationale: Instantiating a full SuperGrid requires the entire supergrid subsystem
  // to query a different table, which is complex. Use a lightweight D3 data-joined table
  // that applies .sg-* classes for visual consistency with SuperGrid's spreadsheet mode.
  renderCatalogTable(datasets);
}
```

```typescript
import * as d3 from 'd3';

function renderCatalogTable(datasets: Array<{...}>): void {
  const bodyEl = dataExplorer?.getCatalogBodyEl();
  if (!bodyEl) return;

  // Clear existing content
  const container = d3.select(bodyEl);
  container.selectAll('*').remove();

  if (datasets.length === 0) {
    // Empty state per UI-SPEC
    const empty = container.append('div').attr('class', 'collapsible-section__stub');
    empty.append('div').attr('class', 'collapsible-section__stub-icon').text('\uD83D\uDDC3');
    const textDiv = empty.append('div').attr('class', 'collapsible-section__stub-text');
    textDiv.append('h3').text('No datasets yet');
    textDiv.append('p').text('Import a file to create your first dataset.');
    return;
  }

  // Build table with sg-* classes for SuperGrid visual consistency
  const table = container.append('table').attr('class', 'dexp-catalog-table');
  const thead = table.append('thead');
  const headerRow = thead.append('tr');
  ['Name', 'Source', 'Cards', 'Connections', 'Last Import'].forEach(h => {
    headerRow.append('th').attr('class', 'sg-header sg-cell').text(h);
  });

  const tbody = table.append('tbody');
  const rows = tbody.selectAll('tr')
    .data(datasets, (d: any) => d.id)
    .enter()
    .append('tr')
    .attr('class', (d: any) => d.is_active ? 'dexp-catalog-row--active' : '')
    .attr('role', 'button')
    .attr('tabindex', '0')
    .attr('aria-selected', (d: any) => d.is_active ? 'true' : 'false')
    .on('click', async (_event: MouseEvent, d: any) => {
      if (d.is_active) return; // No-op on active row
      const confirmed = await AppDialog.show({
        variant: 'confirm',
        title: 'Switch Dataset?',
        message: `Switching to ${d.name} will replace all current data. This cannot be undone.`,
        confirmLabel: 'Switch Dataset',
        cancelLabel: 'Keep Current Dataset',
      });
      if (confirmed) {
        await handleDatasetSwitch(d.id, d.name, d.source_type);
      }
    });

  rows.append('td').attr('class', 'sg-cell').text((d: any) => d.name);
  rows.append('td').attr('class', 'sg-cell').text((d: any) => d.source_type);
  rows.append('td').attr('class', 'sg-cell sg-numeric').text((d: any) => String(d.card_count));
  rows.append('td').attr('class', 'sg-cell sg-numeric').text((d: any) => String(d.connection_count));
  rows.append('td').attr('class', 'sg-cell').text((d: any) => {
    const d2 = new Date(d.last_imported_at);
    return d2.toLocaleDateString();
  });

  // Mark catalog section as ready
  // Find the catalog CollapsibleSection and set state
  bodyEl.classList.add('collapsible-section__body--has-explorer');
}
```

**5. Dataset switch handler** — reuses Phase 85 eviction path pattern from main.ts lines 654-685:

```typescript
async function handleDatasetSwitch(datasetId: string, name: string, sourceType: string): Promise<void> {
  // 1. Show loading state
  viewManager.showLoading();

  // 2. Evict ALL prior data (reuse Phase 85 eviction pattern)
  await sampleManager.evictAll();

  // 3. Reset all provider state to defaults
  filter.resetToDefaults();
  pafv.resetToDefaults();
  selection.clear();
  superPosition.reset();

  // 4. Mark new dataset as active in datasets table
  await bridge.send('db:exec', {
    sql: 'UPDATE datasets SET is_active = 0; UPDATE datasets SET is_active = 1 WHERE id = ?',
    params: [datasetId],
  });

  // 5. Re-introspect schema
  schemaProvider.refresh();

  // 6. Trigger coordinated re-render
  coordinator.scheduleUpdate();

  // 7. Switch to default view
  await viewManager.switchTo('list', () => viewFactory['list']());

  // 8. Refresh Data Explorer display
  void refreshDataExplorer();
}
```

NOTE: The db:exec handler may not support multiple statements separated by `;`. If so, send two separate db:exec calls. Check `src/worker/handlers/` for how db:exec handles SQL. The dataset switch sets is_active=0 on all rows first, then is_active=1 on the target. These are two separate UPDATE statements and should be sent as two bridge.send calls.
