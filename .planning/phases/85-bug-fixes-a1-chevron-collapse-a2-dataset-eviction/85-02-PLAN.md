/**
 * Re-notify all subscribers without changing schema state.
 * Used after dataset eviction — DDL is constant across datasets,
 * so column metadata is unchanged, but subscribers (PropertiesExplorer,
 * ProjectionExplorer, LatchExplorers) need to refresh with reset provider state.
 *
 * Per user decision: "SchemaProvider must re-introspect after new data loads."
 * Since DDL is identical across all datasets (same cards/connections table structure),
 * re-notification achieves the same effect as full PRAGMA re-introspection.
 */
refresh(): void {

}

**Step 3: Rewrite `onLoadSample` callback in main.ts**

Replace the existing `viewManager.onLoadSample = (datasetId: string) => { ... }` block with the full eviction pipeline:

```typescript
viewManager.onLoadSample = (datasetId: string) => {
    void (async () => {
        // 1. Show loading state (prevents flash of old data if eviction takes >100ms)
        viewManager.showLoading?.();

        // 2. Evict ALL prior data (cards + connections)
        await sampleManager.evictAll();

        // 3. Reset all provider state to defaults BEFORE loading new data
        filter.resetToDefaults();
        pafv.resetToDefaults();
        selection.clear();
        superPosition.reset();

        // 4. Load new dataset into freshly emptied database
        await sampleManager.load(datasetId);
        sampleDataLoaded = true;

        // 5. Re-notify SchemaProvider subscribers (per user decision: re-introspect after new data loads)
        //    DDL is constant across datasets, so refresh() re-notifies without re-querying PRAGMA
        schemaProvider.refresh();

        // 6. Trigger coordinated re-render (all views re-query from sql.js)
        coordinator.scheduleUpdate();

        // 7. Navigate to dataset's showcase view
        const dataset = sampleManager.getDatasets().find((d) => d.id === datasetId);
        if (dataset) {
            await viewManager.switchTo(dataset.defaultView, () => viewFactory[dataset.defaultView]());
        }
    })();
};
```

```typescript
/**
 * Isometry v5 — E2E: Dataset Eviction Zero-Bleed
 *
 * Validates: Loading Dataset B after Dataset A fully evicts prior data.
 * No visual bleed — zero cards from Dataset A visible in any view.
 *
 * Requirements: EVIC-01, EVIC-02, EVIC-03
 */

import { test, expect } from '@playwright/test';
import { loadBaselineDataset } from './fixtures';

test.describe('Dataset eviction: zero visual bleed', () => {
    test('switching from meryl-streep to northwind-graph evicts all prior data', async ({ page }) => {
        // 1. Load Dataset A (meryl-streep) via shared baseline
        const initialCount = await loadBaselineDataset(page);
        expect(initialCount).toBeGreaterThan(0);

        // 2. Record a distinctive card name from Dataset A for later zero-bleed check
        const datasetAName = await page.evaluate(async () => {
            const { bridge } = (window as any).__isometry;
            const result = await bridge.send('db:query', {
                sql: "SELECT name FROM cards WHERE name LIKE '%Streep%' OR name LIKE '%Meryl%' LIMIT 1",
                params: [],
            });
            return result.rows[0]?.name ?? null;
        });
        expect(datasetAName).not.toBeNull();

        // 3. Trigger dataset switch to northwind-graph via onLoadSample
        //    (same code path as Command-K selection)
        await page.evaluate(async () => {
            const { viewManager } = (window as any).__isometry;
            // Trigger the full eviction pipeline
            viewManager.onLoadSample('northwind-graph');
        });

        // 4. Wait for new dataset to render (view tab changes or card count changes)
        await page.waitForFunction(
            () => {
                const iso = (window as any).__isometry;
                // Wait until coordinator has processed the update
                return iso && document.querySelectorAll('[data-card-id]').length > 0;
            },
            { timeout: 15_000 },
        );

        // Allow re-render to settle
        await page.waitForTimeout(1000);

        // 5. Assert zero-bleed: no Dataset A cards in database
        const bleedCount = await page.evaluate(async () => {
            const { bridge } = (window as any).__isometry;
            const result = await bridge.send('db:query', {
                sql: "SELECT COUNT(*) as cnt FROM cards WHERE name LIKE '%Streep%' OR name LIKE '%Meryl%'",
                params: [],
            });
            return result.rows[0]?.cnt ?? -1;
        });
        expect(bleedCount).toBe(0);

        // 6. Assert Dataset B cards are present
        const newCount = await page.evaluate(async () => {
            const { bridge } = (window as any).__isometry;
            const result = await bridge.send('db:query', {
                sql: 'SELECT COUNT(*) as cnt FROM cards WHERE deleted_at IS NULL',
                params: [],
            });
            return result.rows[0]?.cnt ?? 0;
        });
        expect(newCount).toBeGreaterThan(0);
        // New dataset should have different count than original
        // (not strictly required, but good sanity check)
    });
});
```
