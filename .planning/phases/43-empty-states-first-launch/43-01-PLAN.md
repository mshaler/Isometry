```typescript
// Wire welcome panel CTAs to import flows
window.addEventListener('isometry:import-file', () => {
  // Trigger file picker — in native mode, post native:action with importFile kind
  // In web mode, create a hidden file input and click it
  if (isNative) {
    // Native: post message to Swift to open file picker
    const nativeBridge = (window as Window & { webkit?: { messageHandlers?: { nativeBridge?: { postMessage: (msg: unknown) => void } } } }).webkit?.messageHandlers?.nativeBridge;
    if (nativeBridge) {
      nativeBridge.postMessage({ type: 'native:request-file-import' });
    }
  } else {
    // Web: create ephemeral file input
    const input = document.createElement('input');
    input.type = 'file';
    input.accept = '.json,.csv,.xlsx,.xls,.md,.html,.htm';
    input.style.display = 'none';
    input.addEventListener('change', async () => {
      const file = input.files?.[0];
      if (!file) return;
      const data = await file.text();
      // Infer source type from extension
      const ext = file.name.split('.').pop()?.toLowerCase() ?? '';
      const sourceMap: Record<string, string> = {
        json: 'json', csv: 'csv', xlsx: 'excel', xls: 'excel',
        md: 'markdown', html: 'html', htm: 'html',
      };
      const source = sourceMap[ext] ?? 'json';
      await bridge.importFile(source as import('./providers').SourceType, data, { filename: file.name });
      coordinator.scheduleUpdate();
    });
    document.body.appendChild(input);
    input.click();
    // Cleanup after selection or cancel
    setTimeout(() => input.remove(), 60000);
  }
});

window.addEventListener('isometry:import-native', () => {
  // Only in native: request Swift to show ImportSourcePickerView
  if (isNative) {
    const nativeBridge = (window as Window & { webkit?: { messageHandlers?: { nativeBridge?: { postMessage: (msg: unknown) => void } } } }).webkit?.messageHandlers?.nativeBridge;
    if (nativeBridge) {
      nativeBridge.postMessage({ type: 'native:show-import-source-picker' });
    }
  }
});
```
