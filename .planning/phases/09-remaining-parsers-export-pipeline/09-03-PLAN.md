```typescript
export class HTMLParser {
  private stripScripts(html: string): string {
    return html
      .replace(/<script\b[^<]*(?:(?!<\/script>)<[^<]*)*<\/script>/gi, '')
      .replace(/<style\b[^<]*(?:(?!<\/style>)<[^<]*)*<\/style>/gi, '')
      // Remove inline event handlers
      .replace(/\s+on\w+="[^"]*"/gi, '')
      .replace(/\s+on\w+='[^']*'/gi, '');
  }

  private extractTitle(html: string): string {
    // Try <title> first
    const titleMatch = html.match(/<title[^>]*>(.*?)<\/title>/i);
    if (titleMatch?.[1]?.trim()) return this.decodeEntities(titleMatch[1].trim());

    // Fallback to <h1>
    const h1Match = html.match(/<h1[^>]*>(.*?)<\/h1>/i);
    if (h1Match?.[1]?.trim()) return this.decodeEntities(h1Match[1].replace(/<[^>]+>/g, '').trim());

    return 'Untitled';
  }

  private extractMeta(html: string, property: string): string | null {
    // Try property="..." format (OpenGraph)
    const propMatch = html.match(new RegExp(`<meta\\s+property=["']${property}["']\\s+content=["']([^"']+)["']`, 'i'));
    if (propMatch?.[1]) return propMatch[1];

    // Try name="..." format
    const nameMatch = html.match(new RegExp(`<meta\\s+name=["']${property}["']\\s+content=["']([^"']+)["']`, 'i'));
    return nameMatch?.[1] ?? null;
  }
}
```

```typescript
private htmlToMarkdown(html: string): string {
  let md = html;

  // Headings (h1-h6)
  md = md.replace(/<h1[^>]*>(.*?)<\/h1>/gi, '# $1\n\n');
  md = md.replace(/<h2[^>]*>(.*?)<\/h2>/gi, '## $1\n\n');
  md = md.replace(/<h3[^>]*>(.*?)<\/h3>/gi, '### $1\n\n');
  md = md.replace(/<h4[^>]*>(.*?)<\/h4>/gi, '#### $1\n\n');
  md = md.replace(/<h5[^>]*>(.*?)<\/h5>/gi, '##### $1\n\n');
  md = md.replace(/<h6[^>]*>(.*?)<\/h6>/gi, '###### $1\n\n');

  // Bold/Italic
  md = md.replace(/<(strong|b)[^>]*>(.*?)<\/\1>/gi, '**$2**');
  md = md.replace(/<(em|i)[^>]*>(.*?)<\/\1>/gi, '*$2*');

  // Links and images
  md = md.replace(/<a\s+href=["']([^"']+)["'][^>]*>(.*?)<\/a>/gi, '[$2]($1)');
  md = md.replace(/<img\s+[^>]*src=["']([^"']+)["'][^>]*alt=["']([^"']*?)["'][^>]*\/?>/gi, '![$2]($1)');
  md = md.replace(/<img\s+[^>]*alt=["']([^"']*?)["'][^>]*src=["']([^"']+)["'][^>]*\/?>/gi, '![$1]($2)');
  md = md.replace(/<img\s+[^>]*src=["']([^"']+)["'][^>]*\/?>/gi, '![]($1)');

  // Code blocks (before inline code)
  md = md.replace(/<pre[^>]*><code[^>]*class=["']language-(\w+)["'][^>]*>([\s\S]*?)<\/code><\/pre>/gi, '```$1\n$2\n```\n');
  md = md.replace(/<pre[^>]*><code[^>]*>([\s\S]*?)<\/code><\/pre>/gi, '```\n$1\n```\n');
  md = md.replace(/<code[^>]*>(.*?)<\/code>/gi, '`$1`');

  // Lists
  md = md.replace(/<li[^>]*>(.*?)<\/li>/gi, '- $1\n');
  md = md.replace(/<\/?[uo]l[^>]*>/gi, '\n');

  // Blockquotes
  md = md.replace(/<blockquote[^>]*>([\s\S]*?)<\/blockquote>/gi, (_, content) => {
    return content.split('\n').map((line: string) => `> ${line.trim()}`).filter(Boolean).join('\n') + '\n';
  });

  // Tables (GFM format)
  md = this.convertTablesToMarkdown(md);

  // Horizontal rules
  md = md.replace(/<hr\s*\/?>/gi, '\n---\n');

  // Paragraphs -> double newlines
  md = md.replace(/<p[^>]*>(.*?)<\/p>/gi, '$1\n\n');
  md = md.replace(/<br\s*\/?>/gi, '\n');

  // Strip remaining tags
  md = md.replace(/<[^>]+>/g, '');

  // Decode HTML entities
  md = this.decodeEntities(md);

  // Clean up whitespace
  md = md.replace(/\n{3,}/g, '\n\n').trim();

  return md;
}

private convertTablesToMarkdown(html: string): string {
  // Convert <table> to GFM table format
  return html.replace(/<table[^>]*>([\s\S]*?)<\/table>/gi, (_, tableContent) => {
    const rows: string[][] = [];
    const rowMatches = tableContent.matchAll(/<tr[^>]*>([\s\S]*?)<\/tr>/gi);

    for (const match of rowMatches) {
      const cells = [...match[1].matchAll(/<t[hd][^>]*>([\s\S]*?)<\/t[hd]>/gi)]
        .map(m => m[1].replace(/<[^>]+>/g, '').trim());
      if (cells.length) rows.push(cells);
    }

    if (rows.length === 0) return '';

    const header = '| ' + rows[0].join(' | ') + ' |';
    const separator = '| ' + rows[0].map(() => '---').join(' | ') + ' |';
    const body = rows.slice(1).map(row => '| ' + row.join(' | ') + ' |').join('\n');

    return `\n${header}\n${separator}\n${body}\n`;
  });
}

private decodeEntities(text: string): string {
  return text
    .replace(/&amp;/g, '&')
    .replace(/&lt;/g, '<')
    .replace(/&gt;/g, '>')
    .replace(/&quot;/g, '"')
    .replace(/&#39;/g, "'")
    .replace(/&nbsp;/g, ' ');
}
```

```typescript
export class HTMLParser {
  parse(htmlStrings: string[], options?: HTMLParseOptions): ParseResult {
    const cards: CanonicalCard[] = [];
    const connections: CanonicalConnection[] = [];
    const errors: ParseError[] = [];
    const personMap = new Map<string, string>(); // name -> cardId

    for (let i = 0; i < htmlStrings.length; i++) {
      try {
        const html = htmlStrings[i];
        const cleaned = this.stripScripts(html);

        // Extract metadata
        const title = this.extractTitle(cleaned);
        const createdAt = this.extractMeta(cleaned, 'article:published_time') ?? new Date().toISOString();
        const sourceUrl = this.extractMeta(cleaned, 'og:url') ?? this.extractCanonical(cleaned);
        const author = this.extractAuthor(cleaned);

        // Convert to Markdown
        const content = this.htmlToMarkdown(cleaned);

        // Create main card
        const mainCardId = crypto.randomUUID();
        const mainCard: CanonicalCard = {
          id: mainCardId,
          card_type: 'note',
          name: title,
          content: content,
          summary: content.slice(0, 200),
          created_at: createdAt,
          modified_at: new Date().toISOString(),
          source: 'html',
          source_id: sourceUrl ?? mainCardId,
          source_url: sourceUrl,
          // ... other fields with defaults
        };
        cards.push(mainCard);

        // Create person card for author
        if (author) {
          let personId = personMap.get(author);
          if (!personId) {
            personId = crypto.randomUUID();
            personMap.set(author, personId);
            cards.push({
              id: personId,
              card_type: 'person',
              name: author,
              source: 'html',
              source_id: `author:${author}`,
              // ... other fields
            });
          }
          connections.push({
            id: crypto.randomUUID(),
            source_id: mainCardId,
            target_id: personId,
            label: 'mentions',
            weight: 1,
            via_card_id: null,
            created_at: new Date().toISOString(),
          });
        }

        // Extract iframe embeds as resource cards
        const iframes = this.extractIframes(cleaned);
        for (const iframe of iframes) {
          const resourceId = crypto.randomUUID();
          cards.push({
            id: resourceId,
            card_type: 'resource',
            name: iframe.title || 'Embedded Content',
            url: iframe.src,
            source: 'html',
            source_id: iframe.src,
            // ... other fields
          });
          connections.push({
            id: crypto.randomUUID(),
            source_id: mainCardId,
            target_id: resourceId,
            label: 'embeds',
            weight: 1,
            via_card_id: null,
            created_at: new Date().toISOString(),
          });
        }

      } catch (error) {
        errors.push({
          index: i,
          source_id: null,
          message: error instanceof Error ? error.message : String(error),
        });
      }
    }

    return { cards, connections, errors };
  }

  private extractAuthor(html: string): string | null {
    // Try meta tags
    const metaAuthor = this.extractMeta(html, 'article:author') ?? this.extractMeta(html, 'author');
    if (metaAuthor) return metaAuthor;

    // Try common byline patterns
    const bylineMatch = html.match(/<[^>]*class=["'][^"']*(?:author|byline)[^"']*["'][^>]*>(.*?)<\/[^>]+>/i);
    if (bylineMatch?.[1]) {
      return bylineMatch[1].replace(/<[^>]+>/g, '').replace(/^By\s+/i, '').trim();
    }

    return null;
  }

  private extractIframes(html: string): Array<{ src: string; title: string }> {
    const iframes: Array<{ src: string; title: string }> = [];
    const matches = html.matchAll(/<iframe[^>]*src=["']([^"']+)["'][^>]*(?:title=["']([^"']*?)["'])?[^>]*>/gi);

    for (const match of matches) {
      iframes.push({ src: match[1], title: match[2] || '' });
    }

    return iframes;
  }
}
```
