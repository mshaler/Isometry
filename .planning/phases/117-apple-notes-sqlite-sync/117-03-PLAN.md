Create `src/etl/apple-notes-direct/validation.ts` with a service for validating data integrity after sync.

**Interface:**

```typescript
export interface ValidationResult {
  valid: boolean;
  errors: ValidationError[];
  warnings: ValidationWarning[];
  stats: ValidationStats;
}

export interface ValidationError {
  type: 'folder_mismatch' | 'timestamp_drift' | 'tag_mismatch' | 'missing_node';
  nodeId: string;
  expected: string;
  actual: string;
  message: string;
}

export interface ValidationWarning {
  type: 'orphan_edge' | 'empty_folder' | 'duplicate_source_id';
  details: string;
}

export interface ValidationStats {
  nodesChecked: number;
  edgesChecked: number;
  folderPathsVerified: number;
  timestampsVerified: number;
  tagsVerified: number;
}

export interface DataIntegrityValidator {
  /**
   * Validate a single node's folder path against expected value
   */
  validateFolderPath(nodeId: string, expectedPath: string): ValidationError | null;

  /**
   * Validate timestamps are within tolerance (default 1 second)
   */
  validateTimestamp(nodeId: string, field: 'created_at' | 'modified_at', expected: Date, toleranceMs?: number): ValidationError | null;

  /**
   * Validate tags match expected array
   */
  validateTags(nodeId: string, expectedTags: string[]): ValidationError | null;

  /**
   * Run full validation suite on all nodes from a source
   */
  validateSource(source: string): ValidationResult;

  /**
   * Validate a specific node by ID
   */
  validateNode(nodeId: string, expected: {
    folder?: string;
    created?: Date;
    modified?: Date;
    tags?: string[];
  }): ValidationError[];
}

export function createDataIntegrityValidator(db: Database): DataIntegrityValidator;
```

**Implementation:**

1. `validateFolderPath`: Query node by ID, compare `folder` field to expected path
2. `validateTimestamp`: Query node, compare timestamp with tolerance (default 1000ms)
3. `validateTags`: Query node, parse `tags` JSON, compare arrays (order-independent)
4. `validateSource`:
   - Query all nodes WHERE `source = ?`
   - Check for orphan edges (edges referencing non-existent nodes)
   - Check for duplicate `source_id` values
   - Return comprehensive ValidationResult
5. `validateNode`: Run folder + timestamp + tags validation for a single node

**Key details:**
- Use sql.js `db.exec()` for all queries
- Parse `tags` column as JSON (it's stored as JSON string)
- Timestamp comparison uses absolute difference <= toleranceMs
- Folder path comparison is exact string match

Update `src/etl/apple-notes-direct/index.ts` to export:

