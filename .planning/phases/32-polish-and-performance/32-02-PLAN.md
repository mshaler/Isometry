const suppressedCollapseKeys = new Set<string>();

**1. Extend `_getCellCardIds()` (line 2763) for aggregate proxy lookup:**

When `findCellInData()` returns a cell with empty `card_ids` (normal for summary cells), check if the cell key corresponds to a summary cell by looking up the collapse state. If the cell's col or row key matches a collapsed group, iterate `_lastCells` to collect card_ids from all child cells under that collapsed group.

```typescript
private _getCellCardIds(cellKey: string): string[] {
  const cell = findCellInData(cellKey, this._lastCells, this._lastRowAxes, this._lastColAxes);
  if (cell?.card_ids?.length) return cell.card_ids;

  // Phase 32 — Aggregate proxy: collect card_ids from child cells under collapsed group
  const { rowKey, colKey } = parseCellKey(cellKey);
  // Check if this is a summary cell position (colKey or rowKey matches a collapsed axis value)
  const collectedIds: string[] = [];
  for (const c of this._lastCells) {
    // Build this cell's full row and col keys
    const cRowKey = this._lastRowAxes.map(a => String((c as Record<string, unknown>)[a.field] ?? 'None')).join(UNIT_SEP);
    const cColKey = this._lastColAxes.map(a => String((c as Record<string, unknown>)[a.field] ?? 'unknown')).join(UNIT_SEP);
    // Match: child cell's key starts with (or equals) the summary cell's key prefix
    const rowMatch = cRowKey === rowKey || cRowKey.startsWith(rowKey + UNIT_SEP);
    const colMatch = cColKey === colKey || cColKey.startsWith(colKey + UNIT_SEP);
    if (rowMatch && colMatch && c.card_ids?.length) {
      collectedIds.push(...c.card_ids);
    }
  }
  return collectedIds;
}
```
