  ```typescript
  setSubtitle(text: string | null): void {
    if (!this._subtitleEl) return;
    if (text === null || text === '') {
      this._subtitleEl.style.display = 'none';
      this._subtitleEl.textContent = '';
    } else {
      this._subtitleEl.style.display = '';
      this._subtitleEl.textContent = text;
    }
  }
  ```
