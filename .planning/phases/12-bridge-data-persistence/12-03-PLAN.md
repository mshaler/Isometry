**1. WKNavigationDelegate for crash recovery (SHELL-05):**

Add `WKNavigationDelegate` conformance to BridgeManager:

```swift
extension BridgeManager: WKNavigationDelegate {
    func webViewWebContentProcessDidTerminate(_ webView: WKWebView) {
        logger.error("WebContent process terminated — initiating recovery")
        showingRecoveryOverlay = true
        isJSReady = false
        // Reload triggers full page load → JS re-initializes → signals native:ready → sendLaunchPayload with last checkpoint
        webView.reload()
    }
}
```

When `native:ready` is received in `didReceive()` (already handled in Plan 01), add:
```swift
case "native:ready":
    isJSReady = true
    showingRecoveryOverlay = false  // Dismiss recovery overlay
    Task { await sendLaunchPayload() }
```

**2. Autosave timer (DATA-05):**

Add to BridgeManager:
```swift
private var autosaveTimer: Timer?

func startAutosave() {
    stopAutosave()
    // Timer.scheduledTimer on main run loop — automatically pauses when app backgrounds
    autosaveTimer = Timer.scheduledTimer(withTimeInterval: 30, repeats: true) { [weak self] _ in
        Task { @MainActor in
            await self?.autosaveTick()
        }
    }
    logger.info("Autosave timer started (30s interval)")
}

func stopAutosave() {
    autosaveTimer?.invalidate()
    autosaveTimer = nil
}

private func autosaveTick() async {
    guard await isDirty else { return }  // Read from DatabaseManager (single source of truth)
    logger.info("Autosave: dirty flag set, requesting checkpoint")
    requestCheckpoint()  // Asks JS to export → JS posts checkpoint → didReceive handles it
}
```

CRITICAL: Use `Timer.scheduledTimer` on the main run loop — NOT `Task.sleep` loop. Timer on main run loop automatically pauses when app backgrounds, satisfying the "timer pauses during background" requirement without extra code. See Pitfall 5 in RESEARCH.md.

**3. saveIfDirty method (for lifecycle events):**

Add to BridgeManager:
```swift
/// Synchronously-safe save path for lifecycle events.
/// Called from scenePhase .background (iOS) and applicationWillTerminate (macOS).
func saveIfDirty() async {
    let dirty = await isDirty
    guard dirty, isJSReady else {
        logger.info("saveIfDirty: skipping (isDirty=\(dirty), isJSReady=\(isJSReady))")
        return
    }
    logger.info("saveIfDirty: requesting checkpoint for lifecycle event")
    requestCheckpoint()
    // Note: This is fire-and-forget. The checkpoint response will come back
    // via didReceive → databaseManager.saveCheckpoint. For background save,
    // we rely on beginBackgroundTask providing ~30s to complete the round-trip.
}
```

Also add a method for the secondary crash detection (webkit bug workaround):
```swift
/// Check if WebContent process crashed silently (webkit bug #176855).
/// Called when scenePhase returns to .active.
func checkForSilentCrash() {
    if webView?.url == nil && !showingRecoveryOverlay {
        logger.warning("Silent WebContent crash detected — webView.url is nil")
        showingRecoveryOverlay = true
        webView?.reload()
    }
}
```

Set webView's `navigationDelegate` to self in the `register(with:)` method or add a `configure(webView:)` method:

struct ContentView: View {
    @StateObject private var bridgeManager = BridgeManager()
    @State private var webView: WKWebView?

    var body: some View {
        ZStack {
            if let webView = webView {
                WebViewContainer(webView: webView)
                    .ignoresSafeArea()

            // Recovery overlay (SHELL-05)
            if bridgeManager.showingRecoveryOverlay {
                recoveryOverlay
            }
        }
        .onAppear {
            setupWebView()
        }
    }

    private var recoveryOverlay: some View {
        ZStack {
            Color.black.opacity(0.5)
                .ignoresSafeArea()
            VStack(spacing: 16) {
                ProgressView()
                    .scaleEffect(1.5)
                Text("Restoring...")
                    .font(.headline)
                    .foregroundColor(.white)
                Text("Your data is safe")
                    .font(.subheadline)
                    .foregroundColor(.white.opacity(0.7))
            }
            .padding(40)
            .background(.ultraThinMaterial, in: RoundedRectangle(cornerRadius: 16))
        }
    }

    private func setupWebView() {
        let config = WKWebViewConfiguration()

        #if os(iOS)
        config.allowsInlineMediaPlayback = true
        #endif

        // Console log forwarding (DEBUG only)
        #if DEBUG
        let consoleScript = WKUserScript(
            source: """
            (function() {
                var origLog = console.log;
                var origWarn = console.warn;
                var origError = console.error;
                function stringify(a) {
                    if (a instanceof Error) return a.message + (a.code ? ' [' + a.code + ']' : '') + (a.stack ? '\\n' + a.stack : '');
                    if (typeof a === 'object' && a !== null) try { return JSON.stringify(a); } catch(e) { return String(a); }
                    return String(a);
                }
                function send(level, args) {
                    try {
                        var parts = [];
                        for (var i = 0; i < args.length; i++) parts.push(stringify(args[i]));
                        window.webkit.messageHandlers.consoleLog.postMessage(level + ': ' + parts.join(' '));
                    } catch(e) {}
                }
                console.log = function() { send('LOG', arguments); origLog.apply(console, arguments); };
                console.warn = function() { send('WARN', arguments); origWarn.apply(console, arguments); };
                console.error = function() { send('ERROR', arguments); origError.apply(console, arguments); };
                window.addEventListener('error', function(e) {
                    send('UNCAUGHT', [e.message, 'at', e.filename + ':' + e.lineno]);
                });
                window.addEventListener('unhandledrejection', function(e) {
                    send('REJECTION', [e.reason instanceof Error ? e.reason.message : String(e.reason)]);
                });
            })();
            """,
            injectionTime: .atDocumentStart,
            forMainFrameOnly: false
        )
        config.userContentController.addUserScript(consoleScript)
        config.userContentController.add(ConsoleLogHandler(), name: "consoleLog")
        #endif

        // Register bridge handler BEFORE creating webView
        bridgeManager.register(with: config)

        // Register URL scheme handler
        config.setURLSchemeHandler(AssetsSchemeHandler(), forURLScheme: "app")

        let wv = WKWebView(frame: .zero, configuration: config)

        #if DEBUG
        wv.isInspectable = true
        #endif

        // Connect BridgeManager to webView
        bridgeManager.configure(webView: wv)

        // Wire DatabaseManager
        do {
            bridgeManager.databaseManager = try DatabaseManager()
        } catch {
            print("[Isometry] Failed to initialize DatabaseManager: \(error)")
            // App will still work but won't persist data
        }

        // Load the web app
        wv.load(URLRequest(url: URL(string: "app://localhost/index.html")!))

        self.webView = wv
    }
}

Rewrite IsometryApp.swift to add lifecycle management:

```swift
import SwiftUI

@main
struct IsometryApp: App {
    // macOS: NSApplicationDelegateAdaptor for applicationWillTerminate (DATA-04)
    // ScenePhase does NOT fire .background on cmd-Q on macOS — this is the only reliable hook
    #if os(macOS)
    @NSApplicationDelegateAdaptor private var appDelegate: IsometryAppDelegate
    #endif

    @Environment(\.scenePhase) private var scenePhase
    @StateObject private var bridgeManager = BridgeManager()

    var body: some Scene {
        WindowGroup {
            ContentView(bridgeManager: bridgeManager)
        }
        .onChange(of: scenePhase) { newPhase in
            handleScenePhaseChange(newPhase)
        }
    }

    private func handleScenePhaseChange(_ newPhase: ScenePhase) {
        switch newPhase {
        case .active:
            // Start autosave timer (DATA-05)
            bridgeManager.startAutosave()
            // Secondary crash detection — webkit bug #176855 workaround
            bridgeManager.checkForSilentCrash()

        case .background:
            // Stop autosave timer
            bridgeManager.stopAutosave()
            // iOS: Save database if dirty (DATA-04)
            #if os(iOS)
            performBackgroundSave()
            #endif

        case .inactive:
            // Transitional state — no action needed
            break

        @unknown default:
            break
        }
    }

    #if os(iOS)
    private func performBackgroundSave() {
        // Request background execution time (~30 seconds)
        let taskID = UIApplication.shared.beginBackgroundTask {
            // Expiration handler — system is about to suspend
            // Nothing to clean up; the save either completed or didn't
        }
        Task {
            await bridgeManager.saveIfDirty()
            // Small delay to allow the checkpoint round-trip (JS export → postMessage → Swift write)
            try? await Task.sleep(nanoseconds: 2_000_000_000) // 2 seconds grace period
            await UIApplication.shared.endBackgroundTask(taskID)
        }
    }
    #endif
}

#if os(macOS)
final class IsometryAppDelegate: NSObject, NSApplicationDelegate {
    /// Set by IsometryApp during initialization
    weak var bridgeManager: BridgeManager?

    func applicationWillTerminate(_ notification: Notification) {
        // macOS: Synchronous-ish save on quit (DATA-04)
        // This is the only reliable termination hook on macOS.
        // We cannot do a full round-trip (ask JS → JS exports → posts back) synchronously.
        // Rely on the last autosave checkpoint (max 30s data loss, accepted per CONTEXT.md).
        // If isDirty, the data since last autosave is lost — this is the accepted tradeoff.
        print("[Isometry] applicationWillTerminate — last autosave checkpoint is the recovery point")
    }
}
#endif
```

- Change `ContentView(bridgeManager:)` init to accept it.

**Option B:** Use @EnvironmentObject.

Go with **Option A** — explicit dependency, no magic.
