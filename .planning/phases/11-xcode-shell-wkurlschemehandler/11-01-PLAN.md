Create `src/main.ts` — the app bootstrap that runs when the page loads. This file:

1. Imports `createWorkerBridge` from `./worker`
2. Imports `ViewManager` from `./views`
3. Imports `StateCoordinator`, `QueryBuilder`, `PAFVProvider`, `FilterProvider`, `SelectionProvider`, `DensityProvider` from `./providers`
4. Imports the 9 view constructors: `ListView`, `GridView`, `KanbanView`, `CalendarView`, `TimelineView`, `GalleryView`, `NetworkView`, `TreeView`, `SuperGrid` from `./views`
5. Imports `ImportToast` from `./ui/ImportToast`

The bootstrap sequence:
```typescript
async function main() {
  const container = document.getElementById('app');
  if (!container) throw new Error('Missing #app container');

  // 1. Create WorkerBridge (initializes sql.js in Worker)
  const bridge = createWorkerBridge();
  await bridge.ready;

  // 2. Initialize the database (creates tables if empty)
  await bridge.send('init-db', null);

  // 3. Create providers
  const filter = new FilterProvider();
  const pafv = new PAFVProvider();
  const selection = new SelectionProvider();
  const density = new DensityProvider();
  const coordinator = new StateCoordinator([filter, pafv, selection, density]);
  const queryBuilder = new QueryBuilder(filter, pafv, density);

  // 4. Create ViewManager
  const viewManager = new ViewManager({
    container,
    coordinator,
    queryBuilder,
    bridge,
    pafv,
  });

  // 5. View factory map
  const viewFactory = {
    list: () => new ListView(container, bridge),
    grid: () => new GridView(container, bridge),
    kanban: () => new KanbanView(container, bridge),
    calendar: () => new CalendarView(container, bridge),
    timeline: () => new TimelineView(container, bridge),
    gallery: () => new GalleryView(container, bridge),
    network: () => new NetworkView(container, bridge),
    tree: () => new TreeView(container, bridge),
    supergrid: () => new SuperGrid(container, bridge),
  };

  // 6. Mount default view (list)
  await viewManager.switchTo('list', viewFactory.list);

  // 7. Expose on window for native bridge (future Phase 12)
  (window as any).__isometry = { bridge, viewManager, viewFactory, pafv, filter, coordinator, queryBuilder };

  console.log('[Isometry] App ready');
}

main().catch(err => console.error('[Isometry] Bootstrap failed:', err));
```
